 DCL     TV_DUBLET       FIXED     (7)       INIT      (0);

 /*********************************************************************/
 /*      HOUSE-KEEPING                                                */
 /*********************************************************************/
         ON ERROR CALL PLIDUMP('SNHOB');

         OPEN FILE (BH657O) TITLE ('BH65700O');

         ON ENDFILE (BH611I)                                            
         BEGIN;                                                         
            EOF = 1;                                                    
         END;                                                           
         CALL ZS38100(DATO1);                                           
         UDDATO = DATO2.DG !! '/' !! DATO2.MD !!                        
                  '-' !! DATO2.ÅR;                                      
         LIN1.DATO = UDDATO;                                            
 /*********************************************************************/
 /*      PROCES                                                       */
 /*********************************************************************/
                                                                        
         READ FILE (BH611I) INTO (BH611_AR);                            
                                                                        
         DO WHILE (EOF = 0 & BH611.EKOMNR < 999);                       
            TV_LÆS = TV_LÆS + 1;                                        
            IF BH611.CINVPNR = '0'   /* KUN VALIDE CPRNR */

               CALL BEHANDL;

         END;                                                           
                                                                        
 /*********************************************************************/
 /*    AFSLUT PROGRAM                                                 */
 /*********************************************************************/

               FILE (BH657O);                                           
                                                                        
 /*********************************************************************/
 /*    OPSÆT STATUS VEDR. 1/7-98 EJERSKAB                             */

         BH657.EEJDNR = BH611.EEJDNR;                                   
         BH657.SORTCPR = BH611.DCPRCIR;                                 
         BH657.DCPRCIR = BH611.DCPRCIR;                                 
                                                                        
         BH657.CÆGTEJET = 0;                                            
         BH657.EJER_NU = '';                                            
                                                                        
         BH657.SORTÅR = 1997;                                           
         BH657.EJER_NU.DANNETÅR = 1997;                                 
         /* 'GAMLE' INDIVIDER SKAL BEVARES MAX. 3 ÅR.  */               
         /* D.V.S. DE DROPPES, HVIS DER IKKE FINDES EN */               
         /* MAKKER (ÆGTEFÆLLE,EFTERLEVENDE ELLER PER-  */               
         /* SONEN SELV) I FORBINDELSE MED OVERFØRSLEN  */               
         /* MELLEM ÆGTEFÆLLER M.V.                     */               
                                                                        
         BH657.EJER_NU.CPRNR = BH611.DCPRCIR;                           
         BH657.EJER_NU.DKØRT = UDDATO;                                  
         BH657.EJER_NU.DATO0107 = 0;                                    
         IF BH611.DAFSTÅ > 0                                            
         THEN                                                           
            BH657.EJER_NU.DEJETTIL = BH611.DAFSTÅ;                      
         ELSE                                                           
            BH657.EJER_NU.DEJETTIL = 99991231;                          
         BH657.EJER_NU.CNEDSL1 = 0;                                     
         BH657.EJER_NU.CNEDSL1GL = 0;                                   
         BH657.EJER_NU.CCIVS = BH611.MT_DATA.CCIVS_MT;                  
         BH657.EJER_NU.DCIÆN = BH611.MT_DATA.DCIÆN;                     
         BH657.EJER_NU.DOVTG = BH611.DOVTG;                             
         BH657.EJER_NU.DAFSTÅ = BH611.DAFSTÅ;                           
         BH657.EJER_NU.DSLUTS = BH611.DSLUTS;                           
         BH657.EJER_NU.DSKØDE = 0;           /* EJ MED FRA ESR I 1997 */
         BH657.EJER_NU.CEJEV = BH611.CEJEV;                             
         BH657.EJER_NU.DVURDÅR = BH611.DVURDÅR;                         
         BH657.EJER_NU.CBENYT = BH611.CBENYT;                           
         BH657.EJER_NU.GL_BEJDV = BH611.BEJDV1996;                      
         BH657.EJER_NU.BBROPFØRT = ' ';      /* EJ MED FRA BBR I 1997 */
                                                                        
         SELECT;                                                        
           WHEN (BH657.EJER_NU.CBENYT = '09' !                          
                 BH657.EJER_NU.CBENYT = '17' !                          
                 BH657.EJER_NU.CBENYT = '70' !                          
                 BH657.EJER_NU.CBENYT = '71')                           
             DO;                                                        
                BH657.EJER_NU.CNEDSL1 = 50;                             
             END;                                                       
           WHEN (BH657.EJER_NU.CBENYT = '  ' &                          
                 BH657.EJER_NU.GL_BEJDV = 0)                            
             DO;                                                        
                BH657.EJER_NU.CNEDSL1 = 50;                             
             END;                                                       
           WHEN (BH657.EJER_NU.DVURDÅR = '1998' &                       
                 BH657.EJER_NU.GL_BEJDV = 0)                            
             DO;                                                        
                BH657.EJER_NU.CNEDSL1 = 50;                             
             END;                                                       
           OTHERWISE                                                    
             CALL OPSÆT_DATO_0107;                                      
         END;                                                           
                                                                        
         BH657.EJER_NU.CNEDSL1GL = BH657.EJER_NU.CNEDSL1;               
                                                                        
         BH657.DHENVN =                                                 
         BH611.MT_DATA.HIST_OPL(1997).DHENVN_MT;                        
                                                                        
         IF (BH657.EJER_NU.DCIÆN >= 19960901 &                          
             BH657.EJER_NU.DCIÆN <= 19971231) &                         
            BH657.EJER_NU.DATO0107 <= BH657.EJER_NU.DCIÆN               
         THEN                                                           
            DO;                                                         
               IF BH611.MT_DATA.HIST_OPL(1997).DHENVN_MT = 0 &          
                  BH611.MT_DATA.HIST_OPL(1996).DHENVN_MT > 0            
               THEN                                                     
                  BH657.DHENVN =                                        
                  BH611.MT_DATA.HIST_OPL(1996).DHENVN_MT;

 /*********************************************************************/
 /*    OPSÆT DATO_0107 P.G.A. SLUTSEDDEL-/OVERTAGELSES- ELLER         */
 /*    SKØDEDATO SAMT OPSÆT KODE FOR EVT. 2 PROMILLE NEDSLAG I EJEN-  */
 /*    DOMSVÆRDISKATTEN                                               */
 /*********************************************************************/
 OPSÆT_DATO_0107:                                                       
 PROC;                                                                  
                                                                        
         SELECT;                                                        
           WHEN (BH657.EJER_NU.DSLUTS > 19980701)                       
              DO;                                                       
                 BH657.EJER_NU.DATO0107 = BH657.EJER_NU.DSLUTS;         
                 BH657.EJER_NU.CNEDSL1 = 30;                            
              END;                                                      
           WHEN (BH657.EJER_NU.DSLUTS > 0 &                             
                 BH657.EJER_NU.DSLUTS <= 19980701)                      
              DO;                                                       
                 BH657.EJER_NU.DATO0107 = BH657.EJER_NU.DSLUTS;         
                 BH657.EJER_NU.CNEDSL1 = 10;                            
              END;                                                      
           WHEN (BH657.EJER_NU.DOVTG > 19980701)                        
              DO;                                                       
                 BH657.EJER_NU.DATO0107 = BH657.EJER_NU.DOVTG;          
                 BH657.EJER_NU.CNEDSL1 = 31;                            
              END;                                                      
           WHEN (BH657.EJER_NU.DOVTG > 0 &                              
                 BH657.EJER_NU.DOVTG <= 19980701)                       
              DO;                                                       
                 BH657.EJER_NU.DATO0107 = BH657.EJER_NU.DOVTG;          
                 BH657.EJER_NU.CNEDSL1 = 11;                            
              END;                                                      
           WHEN (BH657.EJER_NU.DSKØDE > 19980701)                       
              DO;                                                       
                 BH657.EJER_NU.DATO0107 = BH657.EJER_NU.DSKØDE;         
                 BH657.EJER_NU.CNEDSL1 = 32;                            
              END;                                                      
           WHEN (BH657.EJER_NU.DSKØDE > 0 &                             
                 BH657.EJER_NU.DSKØDE <= 19980701)                      
              DO;                                                       
                 BH657.EJER_NU.DATO0107 = BH657.EJER_NU.DSKØDE;         
                 BH657.EJER_NU.CNEDSL1 = 12;                            
              END;                                                      
           OTHERWISE                                                    
              BH657.EJER_NU.CNEDSL1 = 13;                               
         END;                                                           
                                                                        
 END OPSÆT_DATO_0107;                                                   
 /*********************************************************************/
 /*    OPSÆT TYPE OG DATO TIL SORTERING AF OUTPUT-DATASÆT SÅ EVENTUEL */
 /*    OVERFØRSEL AF 'BEDSTE' DATO0107 M.V. KAN FORETAGES             */
 /*    VED TILKØB/FRASALG/TILBAGEKØB/ENKE EFTER SAMT EVT. MELLEM      */
 /*    ÆGTEFÆLLER.                                                    */
 /*********************************************************************/
 OPSÆT_SORT_KRIT:                                                       
 PROC;                                                                  
                                                                        
         BH657.SORTDATO = BH657.EJER_NU.DATO0107;                       
                                                                        
         SELECT;                                                        
           WHEN (BH657.EJER_NU.DAFSTÅ > 0)                              
              DO;                                                       
                 BH657.SORTTYPE = 4;                                    
                 CALL SKRIV_DATASÆT;                                    
              END;                                                      
           WHEN (BH657.EJER_NU.CCIVS = 'D')                             
              DO;                                                       
                 BH657.SORTTYPE = 4;                                    
                 CALL SKRIV_DATASÆT;                                    
              END;                                                      
           OTHERWISE                                                    
              DO;                                                       
                 BH657.SORTTYPE = 6;                                    
                 CALL SKRIV_DATASÆT;                                    
              END;                                                      
         END;                                                           
                                                                        
 END OPSÆT_SORT_KRIT;                                                   
 /*********************************************************************/
 /*    SKRIV DATASÆT                                                  */
 /*********************************************************************/
 SKRIV_DATASÆT:                                                         
 PROC;                                                                  
                                                                        
         IF BH657.EJER_NU.CNEDSL1 = 50                                  
         THEN                                                           
            DO;                                                         
               /* DISSE SKAL FØRST MEDTAGES FRA SLUT 1998 */            
               TV_NED50 = TV_NED50 + 1;                                 
               RETURN;                                                  
            END;                                                        
                                                                        
         /* FØRST SKRIVES AKTUEL PERSON */                              
         WRITE FILE (BH657O) FROM (BH657);                              
         TV_SKRIV = TV_SKRIV + 1;                                       
                                                                        
         /* SÅ SKRIVES EVT. ÆGTEFÆLLE SOM DUBLET */                     
         IF BH657.DHENVN > 0 &                                          
           (BH657.SORTTYPE = 4  !        /* DØDE OG/ELLER SÆLGERE   */  
            BH657.SORTTYPE = 6)          /* IKKE SÆLGERE ELLER DØDE */  
         THEN                                                           
            DO;                                                         
               BH657.SORTTYPE = 3;                       /*DUBLET AF? */
               BH657.SORTCPR = BH657.DHENVN;                            
               WRITE FILE (BH657O) FROM (BH657);                        
               TV_DUBLET = TV_DUBLET + 1;                               
               TV_SKRIV = TV_SKRIV + 1;                                 
            END;                                                        
                                                                        
 END SKRIV_DATASÆT;                                                     
                                                                        
 END BEHANDL;                                                           
 /*********************************************************************/

 DAN_OPT_INDV:                                                          
 PROC;                                                                  
                                                                        
         BH65799.EKOMNR   = 999;                                        
         BH65799.EEJDNR   = 9999999;                                    
         BH65799.SORTCPR  = 99999999999;                                
         BH65799.SORTÅR   = 99999;                                      
         BH65799.SORTDATO = 999999999;                                  
         BH65799.SORTTYPE = 999;                                        
         BH65799.DCPRCIR  = 99999999999;                                
         BH65799.DHENVN   = 99999999999;                                
         BH65799.ANT_INDV = TV_SKRIV;                                   
         WRITE FILE (BH657O) FROM (BH65799);                            
                                                                        
 END DAN_OPT_INDV;                                                      
 /*********************************************************************/
 /*          UDSKRIVNING AF TOTALLISTE                                */
 /*********************************************************************/
 SKRIV_TOTALER:                                                         
 PROC;                                                                  
                                                                        
         CALL ZZ20300(LIN_BLANK,'01');                                  
         CALL ZZ20300(LIN1,'01');                                       
         CALL ZZ20300(LIN2,'01');                                       
         LIN3.TEKST ='ANTAL LÆSTE INDIVIDER PÅ BH61100D BEREGNINGSBÅND';
         LIN3.ANT = TV_LÆS;                                             
         CALL ZZ20300(LIN3,'01');                                       
         IF TV_KOM > 0                                                  
         THEN                                                           
            DO;                                                         
               LIN3.TEKST = 'ANTAL OVERLÆSTE KOMMUNEHEADERE';           
               LIN3.ANT = TV_KOM;                                       
               CALL ZZ20300(LIN3,'01');                                 
            END;                                                        
         LIN3.TEKST = 'ANTAL OVERLÆSTE EJ VALIDE CPRNR';                
         LIN3.ANT = TV_ØVR;                                             
         CALL ZZ20300(LIN3,'01');                                       
         LIN3.TEKST = 'ANTAL OVERLÆSTE CNEDSL1=50     ';                
         LIN3.ANT = TV_NED50;                                           
         CALL ZZ20300(LIN3,'01');                                       
         LIN3.CTL = ZZWS2;                                              
         LIN3.TEKST = 'ANTAL SKREVNE INDIVIDER                        ';
         LIN3.ANT = TV_SKRIV;                                           
         CALL ZZ20300(LIN3,'01');                                       
         LIN3.CTL = ZZWS2;                                              
         LIN3.TEKST = '      HERAF DUBLETTER                          ';
         LIN3.ANT = TV_DUBLET;                                          
         CALL ZZ20300(LIN3,'01');                                       
                                                                        
 END SKRIV_TOTALER;                                                     
                                                                        
                                                                        
 END BH65700;
