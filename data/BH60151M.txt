 /*        BH60151M UNDERSØG GENERELLE UNDTAGELSER VEDR.              */
 /*                 BEREGNING AF LEJEVÆRDI FOR  P E R S O N E N       */
 /*                 FÆLLES MACRO TIL :                                */
 /*                 BH60110 ESKEMA OG BOY                             */
 /*                 BH80110 BOY                                       */
 /*                 BH90110 FORSKUD                                   */
 /* OBS.:  DENNE PROC SKAL KUN AKTIVERES HVIS DER ER BEREGNET DE TRE  */
 /*        LEJEVÆRDIER FOR EJENDOMMEN.                                */
 /*********************************************************************/
 UNDTAGELSER_LEJEVÆRDI_PERSON:                                          
         PROC;                                                          
                                                                        
         BEREGN = NEJ;                                                  
                                                                        
 /*********************************************************************/
 /* OBS.: DER TESTES IKKE FOR BENYTTELSESKODER, GENERELLE UNDTAGELSER */
 /*       SAMT BENYTTELSESKODE-BESTEMTE UNDTAGELSER, SOM ALLEREDE ER  */
 /*       UNDERSØGT VED BEREGNING AF EJENDOMMENS LEJEVÆRDIER          */
 /*********************************************************************/
                                                                        
         IF BH601.CINVPNR ^= '0'            /*  K U N  BEREGNING HVIS */
         THEN                               /*  VALIDT PERSONNUMMER   */
            RETURN;                                                     
                                                                        
         /* DET ER KUN INDIVIDER MED ET VALIDT CPRNR., DER   */         
         /* KAN FÅ BEREGNET EN PERSONLIG LEJEVÆRDI.          */         
                                                                        
         IF BH601.CLEJE_FESR = '6' !             /* UNDTAGELSE  3 */    
            BH601.CLEJE_FESR = '7'                                      
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH601.FEJER > 2                 /* UNDTAGELSE  8 OG */      
         THEN                               /* UNDTAGELSE  10   */      
            DO;                                                         
               IF BH601.CBENYT  = '08' ! BH601.CBENYT = '48'            
               THEN                                                     
                  DO;                                                   
                     IF BH601.CEJ_100 = '1'                             
                     THEN;                                              
                     /* SOMMERHUS OG SAMLET EJERANDEL */                
                     /* ER LIG 100%   -   KAN BEREGNES*/                
                     ELSE                                               
                        RETURN;                                         
                  END;                                                  
               ELSE                                                     
                  IF BH601.FEJER > 3 !                                  
                     BH601.CEJ_100 ^= '1'                               
                  THEN                                                  
                     RETURN;                                            
            END;                                                        
                                                                        
         IF BH601.FEJER = 2                 /* UNDTAGELSE  10   */      
         THEN                                                           
            DO;                                                         
               IF BH601.CEJ_100 = '1' !  /* EJERANDELE IALT = 100 % */  
                  BH601.CEJ_100 = '2'    /* DEN ENE EJERANDEL = O   */  
               THEN;                                                    
               ELSE                                                     
                  RETURN;                                               
            END;                                                        
                                                                        
         IF BH601.FEJER < 2                       /* UNDTAGELSE  ?? */  
         THEN                                                           
            DO;                                                         
               IF BH601.GANE > 0 &              /* EJERANDELEN FOR  */  
               BH601.GANE < 10000               /* EN ENEEJER ER    */  
               THEN                             /* MINDRE END 100 % */  
                  RETURN;                                               
            END;                                                        
                                                                        
         IF MT_CRGN > '01' &               /* UNDTAGELSE  9 - GÆLDER */ 
           (BH601.DOVTG = ÅR0101 !         /* IKKE FOR BOY + FORSKUD */ 
            KØBT_I_ÅR = JA       !         /* HVOR MT_CRGN ER SAT    */ 
            SOLGT_I_ÅR = JA)               /* LIG BLANK              */ 
         THEN                                                           
            RETURN;                                                     
                                                                        
 /***    DER SKAL ALTSÅ FORETAGES LEJEVÆRDIBEREGNING FOR PERSONEN   ***/
         BEREGN = JA;                                                   
                                                                        
 END UNDTAGELSER_LEJEVÆRDI_PERSON;
