 ADVIS_OK: PROC (FEJLNR) RETURNS (BIT(1));                              
                                                                        
 DCL     RETURBIT        BIT       (1);                                 
                                                                        
 DCL     FEJLNR          BIN FIXED (15);                                
                                                                        
         RETURBIT = '0'B;                                               
                                                                        
    /*   IF FEJLNR = 5001 !   DØD */                                    
         IF FEJLNR = 5002 !                                             
            FEJLNR = 5003 !                                             
            FEJLNR = 5004 !                                             
            FEJLNR = 5005 !                                             
            FEJLNR = 5006 !                                             
            FEJLNR = 5007 !                                             
            FEJLNR = 5008 !                                             
            FEJLNR = 5009 !                                             
            FEJLNR = 5010 !                                             
            FEJLNR = 5011 !                                             
            FEJLNR = 5014 !                                             
            FEJLNR = 5015 !                                             
            FEJLNR = 5016 !                                             
            FEJLNR = 5017 !                                             
            FEJLNR = 5019 !                                             
    /*      FEJLNR = 5020 !  SELSKABER */                               
            FEJLNR = 5021 !                                             
            FEJLNR = 5022 !                                             
            FEJLNR = 5024 !                                             
            FEJLNR = 5025 !                                             
            FEJLNR = 5027 !                                             
    /*      FEJLNR = 5028 !                                             
            FEJLNR = 5029 !                                             
            FEJLNR = 5030 !                                             
            FEJLNR = 5045 !                                             
            FEJLNR = 5046 !  UDGÅEDE */                                 
            FEJLNR = 5031 !  /*#01*/                                    
            FEJLNR = 5057 !  /*#01*/                                    
            FEJLNR = 5152 !                                             
            FEJLNR = 5188 !  // D2532                                   
            FEJLNR = 5238 !  // D2679                                   
            FEJLNR = 5240 !  // D2679                                   
            FEJLNR = 5310 !                                             
            FEJLNR = 5314 !                                             
            FEJLNR = 5193    /* KS15-D092 */                            
         THEN                                                           
            RETURBIT = '1'B;                                            
                                                                        
         RETURN (RETURBIT);                                             
                                                                        
 END ADVIS_OK;
