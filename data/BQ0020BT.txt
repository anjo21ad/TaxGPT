 B217_BER_TXT00:PROC(FELTNR) RETURNS (CHAR (1000) VAR);                 
 /*********************************************************************/
 /*    tekster til specifikation af beregning         BQ0020BT        */
 /*    feltnr er identifikation af teksten. Der er ingen betydning    */
 /*    indbygget i en teksts feltnr.                                  */
 /*********************************************************************/
 DCL FELTNR   CHAR (15);                                                
 DCL VAR_TXT  CHAR (1000) VAR;                                          
                                                                        
     SELECT(FELTNR);                                                    
                                                                        
       WHEN ('D1001A') VAR_TXT =                                        
           'Den forskudsregisterede helårlige ' !!                      
           'ejendomsværdiskat er ?UBLBU? kr.';                          
                                                                        
       WHEN ('D1001B') VAR_TXT =                                        
           'Den forskudsregisterede helårlige ' !!                      
           'ejendomsværdiskat';                                         
                                                                        
       WHEN ('D1002A') VAR_TXT =                                        
           'Den forskudsregisterede ejendomsværdiskat for #L#' !!       
           'ejerperioden ?DD/MM? - ?DD/MM? er ?UBLBU? kr.';             
                                                                        
       WHEN ('D1002B') VAR_TXT =                                        
           'Den forskudsregisterede ejendomsværdiskat for ' !!          
           'ejerperioden ?DD/MM? - ?DD/MM?.';                           
                                                                        
       WHEN ('D1003') VAR_TXT =                                         
           'Ejerperiode ?DD/MM? - ?DD/MM? = ?ANTAL? dage. #L#' !!       
           '(?UBLBU? x (?ANTAL? skattepligtige dage : 360))';           
                                                                        
       WHEN ('D1004') VAR_TXT =                                         
           'Forskudsregisterede ejerperiode ?DD/MM? - ?DD/MM?' !!       
           ' = ?ANTAL? dage. Ejerperiode ?DD/MM? - ?DD/MM? = ' !!       
           '?ANTAL? dage. ' !!                                          
           '(?UBLBU? x (?ANTAL? skattepligtige dage : ?ANTAL? ' !!      
           'forskudsregisterede dage))';                                
                                                                        
       WHEN ('D1005') VAR_TXT =                                         
           'Endelig ejendomsværdiskat';                                 
                                                                        
       WHEN ('EJD_SLET') VAR_TXT =                                      
           'Ejendomsværdiskat er ikke beregnet. ' !!                    
           'Ejendom er registeret som slettet';                         
                                                                        
       WHEN ('EJ-EVS') VAR_TXT =                                        
           'Ejendomsværdiskat er ikke beregnet. Ejendom ' !!            
           'er registeret med oplysning om, at der ikke ' !!            
           'skal beregnes ejendomsværdiskat';                           
                                                                        
       WHEN ('EJ-EVS-I-ÅR') VAR_TXT =                                   
           'Ejendomsværdiskat er ikke beregnet. Ejendom ' !!            
           'er registeret med oplysning om, at der ikke ' !!            
           'skal beregnes ejendomsværdiskat i år';                      
                                                                        
       WHEN ('OVERSK1') VAR_TXT =                                       
           '?TXTB? ?UBLBU?';                                            
                                                                        
       WHEN ('OVERSK2') VAR_TXT =                                       
           '?TXTB? ?UBLBU?, samlet Ejendomsværdi ?UBLBU?';              
                                                                        
       WHEN ('UDLEJ-360-DG') VAR_TXT =                                  
           'Ejendomsværdiskat er ikke beregnet. Ejendom ' !!            
           'er registeret som udlejet hele året';                       
                                                                        
       WHEN ('1001') VAR_TXT =                                          
           '10 promille af ?UBLBU?';                                    
                                                                        
       WHEN ('1002') VAR_TXT =                                          
           '10 promille af grundbeløb ?UBLBU?';                         
                                                                        
       WHEN ('1003') VAR_TXT =                                          
           '30 promille af resten ?UBLBU? (?UBLBU? - ?UBLBU?)';         
                                                                        
       WHEN ('1004') VAR_TXT =                                          
           'Erhvervet før 2/7 1998. Nedslag 2 promille af ?UBLBU?';     
                                                                        
       WHEN ('1005') VAR_TXT =                                          
           'Nedslag 4 promille af ?UBLBU?. Max. 1.200 pr. lejlighed';   
                                                                        
       WHEN ('1006') VAR_TXT =                                          
           'Nedslag for 67-årig 4 promille af ?UBLBU?. #L#' !!          
           'Max. 6.000 for helårsbolig';                                
                                                                        
       WHEN ('1007') VAR_TXT =                                          
           'Nedslag for 67-årig 4 promille af ?UBLBU?. #L#' !!          
           'Max. 2.000 for fritidsbolig';                               
                                                                        
       WHEN ('1008') VAR_TXT =                                          
           'Max. stigning til 40 % af overskud 1999 ' !!                
           '(?UBLBU? - ?UBLBU?) + 500. Nedslag ?UBLBU? - ?UBLBU?';      
                                                                        
       WHEN ('1009') VAR_TXT =                                          
           'Max. stigning til 40 % af helårslejeværdi 1999 ' !!         
           '(?UBLBU?) + 500. Nedslag ?UBLBU? - ?UBLBU?';                
                                                                        
       WHEN ('1010') VAR_TXT =                                          
           'Max. stigning til 40 % af overskud 1999 ' !!                
           '(?UBLBU? - ?UBLBU?) + 20 % - 900. ' !!                      
           'Nedslag ?UBLBU? - ?UBLBU?';                                 
                                                                        
       WHEN ('1011') VAR_TXT =                                          
           'Max. stigning til 40 % af helårslejeværdi 1999 ' !!         
           '(?UBLBU?) + 20 % - 900. Nedslag ?UBLBU? - ?UBLBU?';         
                                                                        
       WHEN ('1012') VAR_TXT =                                          
           'Max. stigning til 40 % af overskud 1999 ' !!                
           '(?UBLBU? - ?UBLBU?) + 2.400. Nedslag ?UBLBU? - ?UBLBU?';    
                                                                        
       WHEN ('1013') VAR_TXT =                                          
           'Max. stigning til 40 % af helårslejeværdi 1999 ' !!         
           '(?UBLBU?) + 2.400. Nedslag ?UBLBU? - ?UBLBU?';              
                                                                        
       WHEN ('1014') VAR_TXT =                                          
           'Max. stigning til 40 % af overskud 1999 ' !!                
           '(?UBLBU? - ?UBLBU?) + 20 %. Nedslag ?UBLBU? - ?UBLBU?';     
                                                                        
       WHEN ('1015') VAR_TXT =                                          
           'Max. stigning til 40 % af helårslejeværdi 1999 ' !!         
           '(?UBLBU?) + 20 %. Nedslag ?UBLBU? - ?UBLBU?';               
                                                                        
       WHEN ('1016') VAR_TXT =                                          
           'Max. stigning til 50 % af helårslejeværdi 1999 ' !!         
           '(?UBLBU? - ?UBLBU?) + 2.400. Nedslag ?UBLBU? - ?UBLBU?';    
                                                                        
       WHEN ('1017') VAR_TXT =                                          
           'Max. stigning til 50 % af helårslejeværdi 1999 ' !!         
           '(?UBLBU? - ?UBLBU?) + 20 %. Nedslag ?UBLBU? - ?UBLBU?';     
                                                                        
       WHEN ('1018A') VAR_TXT =                                         
           'Reduktion i nedslag 67-årig. 3 % af indkomst over ' !!      
           'grundbeløb (?UBLBU? - 134.600).';                           
                                                                        
       WHEN ('1018B') VAR_TXT =                                         
           'Reduktion i nedslag 67-årig. 3 % af indkomst over ' !!      
           'grundbeløb (?UBLBU? - 134.600). Max. ?UBLBU?';              
                                                                        
       WHEN ('1019A') VAR_TXT =                                         
           'Reduktion i nedslag 67-årig. 3 % af indkomst over ' !!      
           'grundbeløb (?UBLBU? - 207.000).';                           
                                                                        
       WHEN ('1019B') VAR_TXT =                                         
           'Reduktion i nedslag 67-årig. 3 % af indkomst over ' !!      
           'grundbeløb (?UBLBU? - 207.000). Max. ?UBLBU?';

           'Nedslag for delvis erhverv ?PCTM?% i ?ANTAL? dage #L#' !!   
           '(?PCTM?% af ?UBLBU? x ?ANTAL? dage : 360)';                 
                                                                        
       WHEN ('1024') VAR_TXT =                                          
           'Nedslag for delvis udleje og delvis erhvervsmæssig ' !!     
           'anvendelse ';                                               
                                                                        
       WHEN ('1025') VAR_TXT =                                          
           'Flerårig udlejning i ?ANTAL? dage ' !!                      
           '(?UBLBU? x ?ANTAL? dage : 360)';                            
                                                                        
       WHEN ('1026') VAR_TXT =                                          
           'Nedslag betalt udenlandsk ejendomskat, credit';             
                                                                        
       WHEN ('1027') VAR_TXT =                                          
           'Nedslag betalt udenlandsk ejendomskat, eksemption';         
                                                                        
       WHEN ('1028') VAR_TXT =                                          
           'Ejendomsværdiskat. Ejerandel ?PCTM?% af ?BLBM?';            
                                                                        
       WHEN ('1029') VAR_TXT =                                          
           'Ejendomsværdiskat. Ejerandel 100%';                         
                                                                        
       WHEN ('1029A') VAR_TXT =                                         
           'Ejendomsværdiskat';                                         
                                                                        
       WHEN ('1030') VAR_TXT =                                          
           'Indkomsten vedr. reduktion for 67-årig er opgjort som ' !!  
           'summen af personlig indkomst, positiv kapitalindkomst ' !!  
           'og aktieindkomst, bortset fra aktieindkomst op til ' !!     
           '5.000 kr., hvori der er indeholdt endelig udbytteskat.';    
                                                                        
       WHEN ('1031') VAR_TXT =                                          
           'Indkomsten vedr. reduktion for 67-årig er opgjort som ' !!  
           'summen af ægtefællernes samlede personlige indkomst, ' !!   
           'samlede positive kapitalindkomst ' !!                       
           'og samlede aktieindkomst, ' !!                              
           'bortset fra aktieindkomst op til 10.000 kr., hvori ' !!     
           'der er indeholdt endelig udbytteskat.';                     
                                                                        
       WHEN ('1032') VAR_TXT =                                          
           'Manuelt beregnet ejendomsværdiskat';                        
                                                                        
       WHEN ('1033') VAR_TXT =                                          
           'Samlet ejendomsværdiskat';                                  
                                                                        
       OTHER                                                            
           VAR_TXT = 'MANGLER';                                         
                                                                        
     END; /*SELECT(FELTNR)*/                                            
                                                                        
     RETURN (VAR_TXT);                                                  
                                                                        
 END B217_BER_TXT00;
