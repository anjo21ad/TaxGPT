 PERSONLIG_EJD_LEJEVÆRDI:                                               
       PROC(TAB_LEJEV);                                                 
 DCL   1 TAB_LEJEV(3),                  /* EJENDOMMENS 3 LEJEVÆRDIER  */
         2 BLEJE      FIXED  (11),      /* LEJEV. NORMAL              */
         2 BLEJEFP    FIXED  (11),      /* LEJEV. FØRTIDSPENS./EFTERL.*/
         2 BLEJE67    FIXED  (11);      /* LEJEVÆRDI PENSIONIST       */
                                        /* 1=SIDSTE ÅRS VÆRDIER       */
                                        /* 2=NYT ÅRS VÆRDIER UREDUC.  */
                                        /* 3=NYT ÅRS VÆRDIER REDUC.   */
         /* PÅ HVILKET GRUNDLAG??? */                                   
         IF BH601.CPER_ORD = '3' /* EJER ELL. ÆGTEFÆLLE 67 */           
         THEN                                                           
            DO;                                                         
               IF BH601.PER_LEJEV(1).CREDTYP = 0 /* IKKE REDUCERET */   
               THEN                                                     
                  DO;                                                   
                     BH601.PER_LEJEV(1).BELØB =                         
                     TAB_LEJEV(2).BLEJE67;                              
                     BH601.CLEJVBEKOD = 21;                             
                  END;                                                  
               ELSE                                                     
                  DO;                                                   
                     BH601.PER_LEJEV(1).BELØB =                         
                     TAB_LEJEV(3).BLEJE67;                              
                     BH601.CLEJVBEKOD = 22;                             
                  END;                                                  
               BH601.CREDIG3 = '6';                                     
            END;                                                        
         IF BH601.CPER_ORD = '2' /* EJER ELL. ÆGTEFÆLLE FØRTIDSPENS */  
         THEN                                                           
            DO;                                                         
               IF BH601.PER_LEJEV(1).CREDTYP = 0 /* IKKE REDUCERET */   
               THEN                                                     
                  DO;                                                   
                     BH601.PER_LEJEV(1).BELØB =                         
                     TAB_LEJEV(2).BLEJEFP;                              
                     IF (BH601.CBENYT = '08' !                          
                         BH601.CBENYT = '28' !                          
                         BH601.CBENYT = '48')                           
                     THEN                                               
                        DO;                                             
                           IF (BH601.C_2PCT = '1' !                     
                               BH601.C_2PCT = '2' !                     
                               BH601.C_2PCT = '3')                      
                           THEN                                         
                              BH601.CLEJVBEKOD = 61;                    
                           ELSE                                         
                              BH601.CLEJVBEKOD = 51;                    
                        END;                                            
                     ELSE                                               
                        BH601.CLEJVBEKOD = 11;                          
                  END;                                                  
               ELSE                                                     
                  DO;                                                   
                     BH601.PER_LEJEV(1).BELØB =                         
                     TAB_LEJEV(3).BLEJEFP;                              
                     IF (BH601.CBENYT = '08' !                          
                         BH601.CBENYT = '28' !                          
                         BH601.CBENYT = '48')                           
                     THEN                                               
                        DO;                                             
                           IF (BH601.C_2PCT = '1' !                     
                               BH601.C_2PCT = '2' !                     
                               BH601.C_2PCT = '3')                      
                           THEN                                         
                              DO;                                       
                                 IF BH601.COVG_1998 = '2'               
                                 THEN                                   
                                    BH601.CLEJVBEKOD = 64;              
                                 ELSE                                   
                                    BH601.CLEJVBEKOD = 62;              
                              END;                                      
                           ELSE                                         
                              BH601.CLEJVBEKOD = 52;                    
                        END;                                            
                     ELSE                                               
                        BH601.CLEJVBEKOD = 12;                          
                  END;                                                  
               BH601.CREDIG3 = '1';                                     
            END;                                                        
         IF BH601.CPER_ORD = '1' !  /*IKKE PENSIONIST ELL. FØRTIDSPENS*/
            BH601.CPER_ORD = '4'    /*UNDER 60-ÅRIG PENSIONIST        */
         THEN                                                           
            DO;                                                         
               IF BH601.PER_LEJEV(1).CREDTYP = 0 /* IKKE REDUCERET */   
               THEN                                                     
                  DO;                                                   
                     BH601.PER_LEJEV(1).BELØB =                         
                     TAB_LEJEV(2).BLEJE;                                
                     IF (BH601.CBENYT = '08' !                          
                         BH601.CBENYT = '28' !                          
                         BH601.CBENYT = '48')                           
                     THEN                                               
                        DO;                                             
                           IF (BH601.C_2PCT = '1' !                     
                               BH601.C_2PCT = '2' !                     
                               BH601.C_2PCT = '3')                      
                           THEN                                         
                              BH601.CLEJVBEKOD = 41;                    
                           ELSE                                         
                              BH601.CLEJVBEKOD = 31;                    
                        END;                                            
                     ELSE                                               
                        BH601.CLEJVBEKOD = 01;                          
                  END;                                                  
               ELSE                                                     
                  DO;                                                   
                     BH601.PER_LEJEV(1).BELØB =                         
                     TAB_LEJEV(3).BLEJE;                                
                     IF (BH601.CBENYT = '08' !                          
                         BH601.CBENYT = '28' !                          
                         BH601.CBENYT = '48')                           
                     THEN                                               
                        DO;                                             
                           IF (BH601.C_2PCT = '1' !                     
                               BH601.C_2PCT = '2' !                     
                               BH601.C_2PCT = '3')                      
                           THEN                                         
                              DO;                                       
                                 IF BH601.COVG_1998 = '2'               
                                 THEN                                   
                                    BH601.CLEJVBEKOD = 44;              
                                 ELSE                                   
                                    BH601.CLEJVBEKOD = 42;              
                              END;                                      
                           ELSE                                         
                              BH601.CLEJVBEKOD = 32;                    
                        END;                                            
                     ELSE                                               
                        BH601.CLEJVBEKOD = 02;                          
                  END;                                                  
               BH601.CREDIG3 = '1';                                     
            END;                                                        
 END PERSONLIG_EJD_LEJEVÆRDI;
