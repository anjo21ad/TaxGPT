 fortsættes indtil en ny ejendom dukker op og vi derved har

 Efter sidste fetch(sqlcode100) har vi en samlet ejerkreds              
  for den sidste ejendomm, der skal behandles.                          
                                                                        
 Når vi fetcher ejere tjekkes hhv om                                    
 CEJBEVS er forskellig fra nul og om der er sat FEJLNR 5242

 BEHANDL: PROC;                                                         
                                                                        
                                                                        
 MånedsSum = '';                                                        
                                                                        
                                                                        
 //Vi kigger på ejendomme hvor ejerskab er større end 100               
 //Fordeler ejernes ejerskab ud på de måneder, de har ejet ejendommen   
 //Hvis skilledato er midt i en måned (måneder er delt i to)            
 //registreres ejerskabet på rette del                                  
 //Sælger på første del via TILEVS_DAFSTAA                              
 //Køber på anden del via TILEVS_DOVTG                                  
                                                                        
                                                                        
         K = 1;                                                         
         DO WHILE (K <= Ant_Ejere );                                    
            DATODOV = EJENDOM.TILEVS_DOVTG(k);                          
            DgAFST = EJENDOM.TILEVS_DAFSTAA(k);                         
                                                                        
            //Finder start måned og slut måned for ejerskab             
            FraMd = 1;  //ændres for køber                              
            TilMd = 12; // ændres for sælger                            
                                                                        
            //Hvis overtaget ejendommen i årets løb ændres FraMd        
            //Overtaget dato er ejet fra og med datoen                  
            If DATO_DOV.ÅR = WS_TIL_EVS_AAR Then                        
            Do;                                                         
                   FraMd =  DATO_DOV.MD;                                
                   //overtaget midt i måneden - ejer sidste del         
                   //Hvis lig 1 da ejer hele måneden                    
                   If DATO_DOV.DG > 1 Then                              
                      Del2 = Ja;  //ejer så ikke i første del           
            End;                                                        
            //Hvis afstået ejendommen i årets løb ændres TilMd          
            //Afstået dato er ejet til men ikke med datoen              
            // Hvis 1/1 da ingen ejerskab dette år                      
            If DATO_AFST.ÅR = WS_TIL_EVS_AAR  Then                      
            Do;                                                         
                   TilMd =  DATO_AFST.MD;                               
                   //afstået midt i måneden - ejer første del           
                   //Hvis d.1. da ingen ejerskab i denne måned          
                                                                        
                   //ejer så ikke sidste del                            
                   If DATO_AFST.DG > 1 Then                             
                      Del1 = Ja;                                        
                                                                        
                   //ejer så ikke noget denne md                        
                   If DATO_AFST.DG = 1 & TilMd > 1 Then                 
                      TilMd = TilMd - 1;                                
                                                                        
                   //ejer intet dette år                                
                   If DATO_AFST.DG = 1 & TilMd = 1 Then                 
                      TilMd = 0;                                        
                                                                        
                                                                        
            End;                                                        
                                                                        
                                                                        
                                                                        
            //Finder ejerskab pr måned                                  
            If FraMd <= TilMd Then                                      
            Do;                                                         
               j = FraMd;                                               
               DO WHILE (j <= TilMd );                                  
                   SELECT;                                              
                      //køber ejer kun i del 2                          
                      When ( FraMd = j & Del2)                          
                        Do;                                             
                           MånedsSum.SumMånedDel2(j)                    
                           = MånedsSum.SumMånedDel2(j)                  
                           + EJENDOM.TILEVS_GPCTEJER(k);                
                           If DATO_DOV.DG > 1                           
                              & (DATO_DOV.DG < MånedsSum.DagOver(j)     
                              ! MånedsSum.DagOver(j) = 0) Then          
                              MånedsSum.DagOver(j) = DATO_DOV.DG;       
                        End;                                            
                      //sælger ejer kun i del 1                         
                      When (TilMd = j & Del1)                           
                        Do;                                             
                           MånedsSum.SumMånedDel1(j)                    
                           = MånedsSum.SumMånedDel1(j)                  
                           + EJENDOM.TILEVS_GPCTEJER(k);                
                           If DATO_AFST.DG > 1                          
                              & (DATO_AFST.DG > MånedsSum.DagAFST(j)    
                              ! MånedsSum.DagAFST(j) = 0) Then          
                              MånedsSum.DagAFST(j) = DATO_AFST.DG;      
                        End;                                            
                      //ejerskab hele måneden                           
                      Otherwise                                         
                        Do;                                             
                           MånedsSum.SumMånedDel1(j)                    
                           = MånedsSum.SumMånedDel1(j)                  
                           + EJENDOM.TILEVS_GPCTEJER(k);                
                                                                        
                           MånedsSum.SumMånedDel2(j)                    
                           = MånedsSum.SumMånedDel2(j)                  
                           + EJENDOM.TILEVS_GPCTEJER(k);                
                        End;                                            
                   End;  //Select                                       
                   j = j + 1;                                           
               END;                                                     
            End;                                                        
                                                                        
            Del1 = Nej;                                                 
            Del2 = Nej;                                                 
                                                                        
                                                                        
            K = k+1;                                                    
                                                                        
         END;                                                           
                                                                        
                                                                        
         //Tjekker om der er ejerskab over 100% i en delmåned           
         j = 1;                                                         
         Fundet_Over_100 = NEJ;                                         
         Do While (j <= 12 & ^Fundet_Over_100);                         
                                                                        
            //Hvis dato midt i måneden da en del1 og del2               
            //overlap; der er afstået efter overtaglese                 
            //og begger perioder har ejerskab på 100                    
            //hvis ens datoer da ejer begge ejendommen på dagen.        
            If MånedsSum.DagAFST(j) > 1 &                               
               MånedsSum.DagOver(j) > 1 &                               
               MånedsSum.DagAFST(j) >                                   
               MånedsSum.DagOver(j) &                                   
               (MånedsSum.SumMånedDel1(j) = 100                         
                & MånedsSum.SumMånedDel2(j) = 100) Then                 
            Do;                                                         
               Fundet_Over_100 = Ja;                                    
               SUM_PIC = 195;  //ikke korrekt                           
            End;                                                        
                                                                        
                                                                        
            //En af perioderne er over 100%                             
            If MånedsSum.SumMånedDel1(j) > 100                          
               ! MånedsSum.SumMånedDel2(j) > 100 Then                   
            Do;                                                         
               Fundet_Over_100 = Ja;                                    
               If MånedsSum.SumMånedDel1(j) > 100 Then                  
                  SUM_PIC = MånedsSum.SumMånedDel1(j);                  
               Else SUM_PIC = MånedsSum.SumMånedDel2(j);                
            End;                                                        
                                                                        
            If  Fundet_Over_100 Then                                    
            Do;                                                         
                                                                        
               CALL SKRIV_OVER_100;              //TODO                 
                                                                        
                                                                        
               //to fam. huse advis 5240 ellers 5238                    
               If Ejendom(1).EJERSKAB_CEJDTYPE = '4'                    
                  Then ADVIS_FEJLNR = 5240;                             
                  Else ADVIS_FEJLNR = 5238;                             
                                                                        
                                                                        
               Call UPDATE_RESULT;                                      
            End;                                                        
            j = j + 1;                                                  
         End;                                                           
                                                                        
                                                                        
                                                                        
 End BEHANDL;                                                           
                                                                        
 RÆKKE_FRAVÆLGES:PROC RETURNS(BIT(1));                                  
                                                                        
   /* Nogen rækker frasorters i BE85100 ift. forskellige kriterier. */  
   /* se macro BE85103M betingelser er genskabt her                 */  
   /* De skal ikke tælles op her eller have sat fejlnummer 5242     */  
                                                                        
   DCL    HJ_FRAVÆLGES    BIT       (1);                                
                                                                        
   HJ_FRAVÆLGES  = NEJ;                                                 
   HJ_LOG1       = '';                                                  
                                                                        
                                                                        
   IF DCLBE78000T.TILEVS_A_V_CBENYT = '78'                              
   THEN                                                                 
     DO;                                                                
       HJ_LOG1 = 'FRAVALG_R01';                                         
       RETURN(JA);                                                      
     END;                                                               
                                                                        
   IF DCLBE78000T.TILEVS_A_EJ_COMPLA = '1'
