           2 LÆNGDE BIN FIXED (15),                                     
           %INCLUDE BH61198N;          /* KOPI AF BH60100N FRA 1998 */

           %INCLUDE BH66600N;                                           
 DCL     1 BH66099,                                                     
           %INCLUDE BH66699N;

 /*      DECLARE AF PRINTLINIER                                       */
 /*********************************************************************/

          2 CTL          CHAR      (1),                                 
          2 TX1          CHAR      (37)      INIT                       
            ('PROGRAM: BH66000L'),                                      
          2 TX2          CHAR      (69)      INIT                       
            ('SKAF 1/7-98-STATUS M.V. FRA ORDINÆR ESKEMA 1998'),        
          2 TX3          CHAR      (14)      INIT                       
            ('UDSKREVET DEN '),                                         
          2 DATO         CHAR      (10),                                
          2 REST         CHAR      (02)      INIT      (' ');

          2 F1           CHAR      (54)      INIT      ('  '),          
          2 TX1          CHAR      (20)      INIT                       
                         ('KØRSELSKONTROLLISTE'),

          2 CTL          CHAR      (1),                                 
          2 F1           CHAR      (2)       INIT      (' '),           
          2 TEKST        CHAR      (52),                                
          2 ANT          PIC       'ZZZ.ZZZ.ZZZ.ZZZ.ZZZ',               
          2 REST         CHAR      (59)      INIT      (' ');           
 DCL     1 LIN_BLANK,                                                   
          2 CTL          CHAR      (1),                                 
          2 F1           CHAR      (132)     INIT      (' ');           
         %INCLUDE ZZ20301M;                                             
 /*********************************************************************/
 /*      DECLARE AF TÆLLEVÆRKER                                       */
 /*********************************************************************/
 DCL     TV_LÆS          FIXED     (7)       INIT      (0);             
 DCL     TV_KOM          FIXED     (7)       INIT      (0);             
 DCL     TV_ØVR          FIXED     (7)       INIT      (0);             
 DCL     TV_NED50        FIXED     (7)       INIT      (0);             
 DCL     TV_SKRIV        FIXED     (7)       INIT      (0);             
 DCL     TV_DUBLET       FIXED     (7)       INIT      (0);             
 /*********************************************************************/
 /*      DECLARE AF BUILTIN                                           */
 /*********************************************************************/
 DCL     ADDR            BUILTIN;                                       
 DCL     SUBSTR          BUILTIN;                                       
 DCL     PLIDUMP         BUILTIN;                                       
 /*********************************************************************/
 /*      DECLARE AF DIVERSE FELTER                                    */
 /*********************************************************************/
 DCL     I               BIN FIXED (15);                                
                                                                        
 DCL     UDDATO          CHAR      (10);                                
                                                                        
 DCL     DATO1           CHAR      (26);                                
                                                                        
 DCL     1 DATO2                              BASED(ADDR(DATO1)),       
           2 ÅR          CHAR      (04),                                
           2 F1          CHAR      (01),                                
           2 MD          CHAR      (02),                                
           2 F2          CHAR      (01),                                
           2 DG          CHAR      (02);

 /*********************************************************************/
 /*      HOUSE-KEEPING                                                */
 /*********************************************************************/
         ON ERROR CALL PLIDUMP('SNHOB');                                
         CALL ZS61900;                 /*   UDSKRIVNING AF SSI-DATO   */
         LIN_BLANK.CTL = ZZIK1;                                         
         LIN1.CTL = ZZWS2;                                              
         LIN2.CTL = ZZWS3;                                              
         LIN3.CTL = ZZWS2;                                              
         OPEN FILE (BH611I) TITLE ('BH61100I');                         
         OPEN FILE (BH660O) TITLE ('BH66000O');                         
         CALL ZZ20390 ('*OPEN','BH66001Y');                             
         ON ENDFILE (BH611I)                                            
         BEGIN;                                                         
            EOF = 1;                                                    
         END;                                                           
         CALL ZS38100(DATO1);                                           
         UDDATO = DATO2.DG !! '/' !! DATO2.MD !!                        
                  '-' !! DATO2.ÅR;                                      
         LIN1.DATO = UDDATO;                                            
 /*********************************************************************/
 /*      PROCES                                                       */
 /*********************************************************************/
                                                                        
         READ FILE (BH611I) INTO (BH611_AR);                            
                                                                        
         DO WHILE (EOF = 0 & BH611.EKOMNR < 999);                       
            TV_LÆS = TV_LÆS + 1;                                        
            IF BH611.EKOMNR = 0         /* KOMMUNEINDIVIDER */          
            THEN                                                        
               TV_KOM = TV_KOM + 1;                                     
            ELSE                                                        
               IF BH611.CINVPNR = '0'   /* KUN VALIDE CPRNR */          
               THEN                     /* SKAL BEHANDLES   */          
                  CALL BEHANDL;                                         
               ELSE                                                     
                  TV_ØVR = TV_ØVR + 1;  /* SE-NR./LØBENR.   */          
            READ FILE (BH611I) INTO (BH611_AR);                         
         END;                                                           
                                                                        
 /*********************************************************************/
 /*    AFSLUT PROGRAM                                                 */
 /*********************************************************************/
                                                                        
         CALL DAN_OPT_INDV;                                             
                                                                        
         CALL SKRIV_TOTALER;                                            
                                                                        
         CALL ZZ20300(LIN_BLANK,'99');                                  
                                                                        
         CLOSE FILE (BH611I),                                           
               FILE (BH660O);                                           
                                                                        
 /*********************************************************************/
 /*    OPSÆT STATUS VEDR. 1/7-98 EJERSKAB                             */
 /*********************************************************************/
 BEHANDL:                                                               
 PROC;                                                                  
                                                                        
         BH660.EKOMNR = BH611.EKOMNR;                                   
         BH660.EEJDNR = BH611.EEJDNR;                                   
         BH660.SORTCPR = BH611.DCPRCIR;                                 
         BH660.DCPRCIR = BH611.DCPRCIR;                                 
                                                                        
         BH660.CÆGTEJET = 0;                                            
         BH660.EJER_NU = '';                                            
                                                                        
         BH660.EJER_NU.DANNETÅR = 1998;                                 
         BH660.SORTÅR = 1998;                                           
         /* 'GAMLE' INDIVIDER SKAL BEVARES MAX. 3 ÅR.  */               
         /* D.V.S. DE DROPPES, HVIS DER IKKE FINDES EN */               
         /* MAKKER (ÆGTEFÆLLE,EFTERLEVENDE ELLER PER-  */               
         /* SONEN SELV) I FORBINDELSE MED OVERFØRSLEN  */               
         /* MELLEM ÆGTEFÆLLER M.V.                     */               
                                                                        
         BH660.EJER_NU.CPRNR = BH611.DCPRCIR;                           
         BH660.EJER_NU.DKØRT = UDDATO;                                  
         BH660.EJER_NU.DATO0107 = 0;                                    
         IF BH611.DAFSTÅ > 0                                            
         THEN                                                           
            BH660.EJER_NU.DEJETTIL = BH611.DAFSTÅ;                      
         ELSE                                                           
            BH660.EJER_NU.DEJETTIL = 99991231;                          
         BH660.EJER_NU.CNEDSL1 = 0;                                     
         BH660.EJER_NU.CNEDSL1GL = 0;                                   
         BH660.EJER_NU.CCIVS = BH611.MT_DATA.CCIVS_MT;                  
         BH660.EJER_NU.DCIÆN = BH611.MT_DATA.DCIÆN;                     
         BH660.EJER_NU.DOVTG = BH611.DOVTG;                             
         BH660.EJER_NU.DAFSTÅ = BH611.DAFSTÅ;                           
         BH660.EJER_NU.DSLUTS = BH611.DSLUTS;                           
         BH660.EJER_NU.DSKØDE = BH611.DSKØDE;                           
         BH660.EJER_NU.CEJEV = BH611.CEJEV;                             
         BH660.EJER_NU.DVURDÅR = BH611.DVURDÅR;                         
         BH660.EJER_NU.CBENYT = BH611.CBENYT;                           
         BH660.EJER_NU.GL_BEJDV = BH611.GL_BEJDV;                       
         BH660.EJER_NU.BBROPFØRT = ' '; /* FØRST MED I ORD 1999 */      
                                                                        
         SELECT;                                                        
           WHEN (BH660.EJER_NU.CBENYT = '09' !                          
                 BH660.EJER_NU.CBENYT = '17' !                          
                 BH660.EJER_NU.CBENYT = '70' !                          
                 BH660.EJER_NU.CBENYT = '71')                           
             DO;                                                        
                BH660.EJER_NU.CNEDSL1 = 50;                             
             END;                                                       
           WHEN (BH660.EJER_NU.CBENYT = '  ' &                          
                 BH660.EJER_NU.GL_BEJDV = 0)                            
             DO;                                                        
                BH660.EJER_NU.CNEDSL1 = 50;                             
             END;                                                       
           WHEN (BH660.EJER_NU.DVURDÅR = '1999' &                       
                 BH660.EJER_NU.GL_BEJDV = 0)                            
             DO;                                                        
                BH660.EJER_NU.CNEDSL1 = 50;                             
             END;                                                       
           OTHERWISE                                                    
             CALL OPSÆT_DATO_0107;                                      
         END;                                                           
                                                                        
         BH660.EJER_NU.CNEDSL1GL = BH660.EJER_NU.CNEDSL1;               
                                                                        
         BH660.DHENVN =                                                 
         BH611.MT_DATA.HIST_OPL(1998).DHENVN_MT;                        
                                                                        
         IF (BH660.EJER_NU.DCIÆN >= 19970901 &                          
             BH660.EJER_NU.DCIÆN <= 19981231) &                         
            BH660.EJER_NU.DATO0107 <= BH660.EJER_NU.DCIÆN               
         THEN                                                           
            DO;                                                         
               IF BH611.MT_DATA.HIST_OPL(1998).DHENVN_MT = 0 &          
                  BH611.MT_DATA.HIST_OPL(1997).DHENVN_MT > 0            
               THEN                                                     
                  BH660.DHENVN =                                        
                  BH611.MT_DATA.HIST_OPL(1997).DHENVN_MT;               
               /* EJENDOMMEN VAR OGSÅ EJET AF PERSONEN FØR ELLER PÅ */  
               /* DEN SENESTE CIVILSTANDSÆNDRINGSDATO               */  
               BH660.CÆGTEJET = 1;                                      
            END;                                                        
                                                                        
         BH660.EJER_FØR = '';                                           
                                                                        
         CALL OPSÆT_SORT_KRIT;                                          
                                                                        
 /*********************************************************************/
 /*    OPSÆT DATO_0107 P.G.A. SLUTSEDDEL-/OVERTAGELSES- ELLER         */
 /*    SKØDEDATO SAMT OPSÆT KODE FOR EVT. 2 PROMILLE NEDSLAG I EJEN-  */
 /*    DOMSVÆRDISKATTEN                                               */
 /*********************************************************************/
 OPSÆT_DATO_0107:                                                       
 PROC;                                                                  
                                                                        
         SELECT;                                                        
           WHEN (BH660.EJER_NU.DSLUTS > 19980701)                       
              DO;                                                       
                 BH660.EJER_NU.DATO0107 = BH660.EJER_NU.DSLUTS;         
                 BH660.EJER_NU.CNEDSL1 = 30;                            
              END;                                                      
           WHEN (BH660.EJER_NU.DSLUTS > 0 &                             
                 BH660.EJER_NU.DSLUTS <= 19980701)                      
              DO;                                                       
                 BH660.EJER_NU.DATO0107 = BH660.EJER_NU.DSLUTS;         
                 BH660.EJER_NU.CNEDSL1 = 10;                            
              END;                                                      
           WHEN (BH660.EJER_NU.DOVTG > 19980701)                        
              DO;                                                       
                 BH660.EJER_NU.DATO0107 = BH660.EJER_NU.DOVTG;          
                 BH660.EJER_NU.CNEDSL1 = 31;                            
              END;                                                      
           WHEN (BH660.EJER_NU.DOVTG > 0 &                              
                 BH660.EJER_NU.DOVTG <= 19980701)                       
              DO;                                                       
                 BH660.EJER_NU.DATO0107 = BH660.EJER_NU.DOVTG;          
                 BH660.EJER_NU.CNEDSL1 = 11;                            
              END;                                                      
           WHEN (BH660.EJER_NU.DSKØDE > 19980701)                       
              DO;                                                       
                 BH660.EJER_NU.DATO0107 = BH660.EJER_NU.DSKØDE;         
                 BH660.EJER_NU.CNEDSL1 = 32;                            
              END;                                                      
           WHEN (BH660.EJER_NU.DSKØDE > 0 &                             
                 BH660.EJER_NU.DSKØDE <= 19980701)                      
              DO;                                                       
                 BH660.EJER_NU.DATO0107 = BH660.EJER_NU.DSKØDE;         
                 BH660.EJER_NU.CNEDSL1 = 12;                            
              END;                                                      
           OTHERWISE                                                    
              BH660.EJER_NU.CNEDSL1 = 13;                               
         END;                                                           
                                                                        
 END OPSÆT_DATO_0107;                                                   
 /*********************************************************************/
 /*    OPSÆT TYPE OG DATO TIL SORTERING AF OUTPUT-DATASÆT SÅ EVENTUEL */
 /*    OVERFØRSEL AF 'BEDSTE' DATO0107 M.V. KAN FORETAGES             */
 /*    VED TILKØB/FRASALG/TILBAGEKØB/ENKE EFTER SAMT EVT. MELLEM      */
 /*    ÆGTEFÆLLER.                                                    */
 /*********************************************************************/
 OPSÆT_SORT_KRIT:                                                       
 PROC;                                                                  
                                                                        
         BH660.SORTDATO = BH660.EJER_NU.DATO0107;                       
                                                                        
         SELECT;                                                        
           WHEN (BH660.EJER_NU.DAFSTÅ > 0)                              
              DO;                                                       
                 BH660.SORTTYPE = 4;                                    
                 CALL SKRIV_DATASÆT;                                    
              END;                                                      
           WHEN (BH660.EJER_NU.CCIVS = 'D')                             
              DO;                                                       
                 BH660.SORTTYPE = 4;                                    
                 CALL SKRIV_DATASÆT;                                    
              END;                                                      
           OTHERWISE                                                    
              DO;                                                       
                 BH660.SORTTYPE = 6;                                    
                 CALL SKRIV_DATASÆT;                                    
              END;                                                      
         END;                                                           
                                                                        
 END OPSÆT_SORT_KRIT;                                                   
 /*********************************************************************/
 /*    SKRIV DATASÆT                                                  */
 /*********************************************************************/
 SKRIV_DATASÆT:                                                         
 PROC;                                                                  
                                                                        
         /* FØRST SKRIVES AKTUEL PERSON */                              
         WRITE FILE (BH660O) FROM (BH660);                              
         TV_SKRIV = TV_SKRIV + 1;                                       
                                                                        
         /* SÅ SKRIVES EVT. ÆGTEFÆLLE SOM DUBLET */                     
         IF BH660.DHENVN > 0 &                                          
           (BH660.SORTTYPE = 4  !        /* DØDE OG/ELLER SÆLGERE   */  
            BH660.SORTTYPE = 6)          /* IKKE SÆLGERE ELLER DØDE */  
         THEN                                                           
            DO;                                                         
               BH660.SORTTYPE = 3;                       /*DUBLET AF? */
               BH660.SORTCPR = BH660.DHENVN;                            
               WRITE FILE (BH660O) FROM (BH660);                        
               TV_DUBLET = TV_DUBLET + 1;                               
               TV_SKRIV = TV_SKRIV + 1;                                 
            END;                                                        
                                                                        
 END SKRIV_DATASÆT;                                                     
                                                                        
 END BEHANDL;                                                           
 /*********************************************************************/
 /*      DER DANNES ET SLUTINDIVID TIL BRUG FOR MASKINEL AFSTEMNING   */
 /*********************************************************************/
 DAN_OPT_INDV:                                                          
 PROC;                                                                  
                                                                        
         BH66099.EKOMNR   = 999;                                        
         BH66099.EEJDNR   = 9999999;                                    
         BH66099.SORTCPR  = 99999999999;                                
         BH66099.SORTÅR   = 99999;                                      
         BH66099.SORTDATO = 999999999;                                  
         BH66099.SORTTYPE = 999;                                        
         BH66099.DCPRCIR  = 99999999999;                                
         BH66099.DHENVN   = 99999999999;                                
         BH66099.ANT_INDV = TV_SKRIV;                                   
         WRITE FILE (BH660O) FROM (BH66099);                            
                                                                        
 END DAN_OPT_INDV;                                                      
 /*********************************************************************/
 /*          UDSKRIVNING AF TOTALLISTE                                */
 /*********************************************************************/
 SKRIV_TOTALER:                                                         
 PROC;                                                                  
                                                                        
         CALL ZZ20300(LIN_BLANK,'01');                                  
         CALL ZZ20300(LIN1,'01');                                       
         CALL ZZ20300(LIN2,'01');                                       
         LIN3.TEKST ='ANTAL LÆSTE INDIVIDER PÅ BH61100D BEREGNINGSBÅND';
         LIN3.ANT = TV_LÆS;                                             
         CALL ZZ20300(LIN3,'01');                                       
         IF TV_KOM > 0                                                  
         THEN                                                           
            DO;                                                         
               LIN3.TEKST = 'ANTAL OVERLÆSTE KOMMUNEHEADERE';           
               LIN3.ANT = TV_KOM;                                       
               CALL ZZ20300(LIN3,'01');                                 
            END;                                                        
         LIN3.TEKST = 'ANTAL OVERLÆSTE EJ VALIDE CPRNR';                
         LIN3.ANT = TV_ØVR;                                             
         CALL ZZ20300(LIN3,'01');                                       
         LIN3.TEKST = 'ANTAL OVERLÆSTE CNEDSL1=50     ';                
         LIN3.ANT = TV_NED50;                                           
         CALL ZZ20300(LIN3,'01');                                       
         LIN3.CTL = ZZWS2;                                              
         LIN3.TEKST = 'ANTAL SKREVNE INDIVIDER                        ';
         LIN3.ANT = TV_SKRIV;                                           
         CALL ZZ20300(LIN3,'01');                                       
         LIN3.CTL = ZZWS2;                                              
         LIN3.TEKST = '      HERAF DUBLETTER                          ';
         LIN3.ANT = TV_DUBLET;                                          
         CALL ZZ20300(LIN3,'01');                                       
                                                                        
 END SKRIV_TOTALER;                                                     
                                                                        
                                                                        
 END BH66000;
