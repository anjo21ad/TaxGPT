         EXTERNAL INIT ('NOTEST(ALL,,;)');

 DCL    1 SIDELIN,                                                      
          2 CCA          CHAR      (01),                                
          2 FILLER       CHAR      (132)     INIT      (' ');           
                                                                        
 DCL     1 OVERLIN1,                                                    
           2 CCA         CHAR      (01),                                
           2 TEKST1      CHAR      (17)      INIT      (                
           'PROGRAM: BH64200L'),                                        
           2 F1          CHAR      (25)      INIT      (' '),           
           2 TEKST2      CHAR      (57)      INIT      (                
           'BEHANDLING AF EVT. HELÅRSBEBOEDE SOMMERHUSE   '),           
           2 F2          CHAR      (12)      INIT      (' '),           
           2 TEKST3      CHAR      (11)      INIT      (                
           'UDSKREVET:'),                                               
           2 DATO,                                                      
             3 DAG         CHAR      (02),                              
             3 F1          CHAR      (01) INIT ('.'),                   
             3 MD          CHAR      (02),

           2 CCA         CHAR      (01),                                
           2 F1          CHAR      (46)      INIT      (' '),           
           2 TEKST1      CHAR      (86)      INIT                       
           ('          KØRSELSKONTROLLISTE');                           
 DCL     1 DETLIN0,                                                     
           2 CCA         CHAR      (01),                                
           2 TEKST1      CHAR      (41),                                
           2 CPUÅR       PIC       '99',                                
           2 TEKST2      CHAR      (11),                                
           2 START       PIC       '999999',                            
           2 TEKST3      CHAR      (9),                                 
           2 SLUT        PIC       '999999',                            
           2 TEKST4      CHAR      (57);                                
 DCL     1 DETLIN1,                                                     
           2 CCA         CHAR      (01),                                
           2 TEKST       CHAR      (60),                                
           2 ANTAL       PIC       'ZZZ.ZZZ.ZZ9',                       
           2 F2          CHAR      (61)      INIT      (' ');           
 DCL     1 DETLIN2,                                                     
           2 CCA         CHAR      (01),                                
           2 TEKST       CHAR      (33),                                
           2 RCKOD       CHAR      (01)      INIT      (' '),           
           2 F1          CHAR      (19)      INIT      (' '),           
           2 ANTAL       PIC       'ZZZ.ZZZ.ZZ9',                       
           2 F2          CHAR      (68)      INIT      (' ');           
         % INCLUDE ZZ20301M;                                            
 /********************************************************************* 
 *       HOUSEKEEPING                                                 * 
 *********************************************************************/ 
                                                                        
         CALL ZS61900;                          /* UDSKRIVNING AF SSI */
                                                                        
         EOF_CN570 = '0'B;                                              
         EOF_BH655 = '0'B;                                              
                                                                        
         ON ERROR                                                       
            CALL PLIDUMP ('SNHOB');                                     
                                                                        
         OPEN FILE (CN570I)  TITLE ('CN57100I');                        
         OPEN FILE (BH655I)  TITLE ('BH65501I');                        
         OPEN FILE (BH642O)  TITLE ('BH64200O');                        
                                                                        
         ON ENDFILE (CN570I)                                            
            EOF_CN570 = '1'B;                                           
                                                                        
         ON ENDFILE (BH655I)                                            
            EOF_BH655 = '1'B;                                           
                                                                        
         CALL ZZ20390 ('*OPEN','BH64201Y');                             
                                                                        
         CALL ZS38100(DATO1);                                           
         OVERLIN1.DATO = DATO2, BY NAME;                                
         SIDELIN.CCA   = ZZIK1;                                         
         OVERLIN1.CCA  = ZZWS2;                                         
         OVERLIN2.CCA  = ZZWS3;                                         
                                                                        
         TV = 0;                                                        
                                                                        
         /* OPSÆT KØRSELSTYPE UD FRA KØRSELSMÅNEDEN */                  
                                                                        
         IF DATO2.MD < '06' !                                           
            DATO2.MD > '10'                                             
         THEN                                                           
            KØRSEL = 'ESKEMA';                                          
         ELSE                                                           
            KØRSEL = '      ';                                          
                                                                        
 /********************************************************************* 
 *       HOVEDSTYRING                                                 * 
 *********************************************************************/ 
                                                                        
         CALL LÆS_MTFIL;                                                
                                                                        
         CALL LÆS_CN570;                                                
                                                                        
         DO WHILE ( ^EOF_CN570 );                                       
            SELECT;                                                     
               WHEN ( MTFIL.DCPRN < CN570.DCPRCIR & ^EOF_BH655 )        
                  CALL LÆS_MTFIL;                                       
               OTHERWISE                                                
                  DO;                                                   
                     CALL BEHANDL;   /* BEHANDLING AF ESR-UDTRÆK */     
                     CALL LÆS_CN570;                                    
                  END;                                                  
            END;                                                        
         END;                                                           
                                                                        
         DO WHILE ( ^EOF_BH655 );                                       
            CALL LÆS_MTFIL;                                             
         END;                                                           
                                                                        
         CALL AFSLUT;                                                   
                                                                        
 /********************************************************************* 
 *       LÆS UDTRÆK FRA ESR                                           * 
 *********************************************************************/ 
                                                                        
 LÆS_CN570:                                                             
         PROC;                                                          
                                                                        
         READ FILE (CN570I) SET (CN570_PTR);                            
                                                                        
         IF CN57099.DCPRCIR = 9999999999     /* OPTÆLLINGSINDIVID */    
         THEN                                                           
            DO;                                                         
               DO WHILE ( ^EOF_CN570 );                                 
                  INDEX = INDEX + 1;                                    
                  TAB.CRCK(INDEX) = CN57099.CRCK;                       
                  TAB.ANT_INDV(INDEX) = CN57099.ANT_INDV;               
                  READ FILE (CN570I) SET (CN570_PTR);                   
               END;                                                     
            END;                                                        
                                                                        
         IF ^EOF_CN570                                                  
         THEN                                                           
            TV.LÆS_CN570 = TV.LÆS_CN570 + 1;                            
                                                                        
 END LÆS_CN570;                                                         
 /*********************************************************************/
 /*      LÆS MINI-MTFIL B.H65501D                                     */
 /*********************************************************************/
                                                                        
 LÆS_MTFIL:                                                             
         PROC;                                                          
                                                                        
         READ FILE (BH655I) INTO (MTFIL);                               
                                                                        
         IF ^EOF_BH655 &                                                
            MTFIL_OPTÆL.DCPRN ^= 99999999999 /* OPTÆLLINGSINDV.*/       
         THEN                                                           
            TV.LÆS_MTFIL = TV.LÆS_MTFIL + 1;                            
                                                                        
         IF MTFIL_OPTÆL.DCPRN = 99999999999 /* OPTÆLLINGSINDV.*/        
         THEN                                                           
            READ FILE (BH655I) INTO (MTFIL);                            
                                                                        
 END LÆS_MTFIL;                                                         
 /********************************************************************* 
 *       PERSONER SOM EVT. SKAL BEHANDLES/BEREGNES                    * 
 *********************************************************************/ 
                                                                        
 BEHANDL:                                                               
         PROC;                                                          
                                                                        
         IF CN570.CBENYT = '08' !                                       
            CN570.CBENYT = '28' !                                       
            CN570.CBENYT = '48'                                         
         THEN;                                                          
         ELSE                                                           
            DO;                                                         
               /* FORKERT INPUT TIL KØRSLEN !!! */                      
               CALL ZS30101(4,'BH64200 ** 251 ' !!                      
               'FORKERT ESR-UDTRÆK BENYTTET (=C.N571X1D?)',2);          
            END;                                                        
                                                                        
         IF (CN570.DAFGDAT > 0 &        /* HVIS EJEREN ER   */          
             CN570.DAFSTÅ = 0)          /* SLETTET I ESR    */          
         THEN                           /* SPRINGES OVER    */          
            DO;                                                         
               /* DETTE ER EJENDOMME SOM JVF.   */                      
               /* SNITFLADEN FOR ESR-UDTRÆKKET  */                      
               /*  I K K E  BURDE FOREKOMME HER */                      
               CALL SKRIV_UDTRÆK;                                       
               RETURN;                                                  
            END;                                                        
                                                                        
         IF (CN570.DAFSTÅ > 0 &         /* SALG FØR INDKOMSTÅRET */     
             CN570.DAFSTÅ < ÅR0101) !   /* SPRINGES OVER         */     
            CN570.DOVTG > ÅR1231        /* KØB EFTER INDKOMSTÅRET */    
         THEN                                                           
            DO;                                                         
               /* DETTE ER EJENDOMME SOM JVF.   */                      
               /* SNITFLADEN FOR ESR-UDTRÆKKET  */                      
               /*  I K K E  BURDE FOREKOMME HER */                      
               CALL SKRIV_UDTRÆK;                                       
               RETURN;                                                  
            END;                                                        
                                                                        
         /* KONTROLLER STATUS FOR EJERS CPRNR, D.V.S. OM DER ER   */    
         /* TALE OM ET RIGTIGT CPRNR, ET SE-NR, ET INVALIDT CPRNR */    
         /* ELLER ET LØBENR.                                      */    
                                                                        
         HJ_CPR = CN570.DCPRCIR;                                        
         CALL DAN_CPRNR_STATUS(HJ_CPR,CN570.CEJFORH,CRETUR);            
                                                                        
         IF CRETUR ^= '0'                  /* KUN HVIS      */          
         THEN                              /* VALIDT CPRNR  */          
            DO;                            /* FORETAGES     */          
               CALL SKRIV_UDTRÆK;          /* BEHANDLING    */          
               RETURN;                     /* ELLERS        */          
            END;                           /* SPRINGES OVER */          
                                                                        
         IF CN570.BOPÆLSKODE < '1'                                      
         THEN                                                           
            DO;                                                         
               /* EJEREN BOR IKKE PÅ ADRESSEN JVF. P-DATA */            
               /* D.V.S. INGEN BEHANDLING!                */            
               CALL SKRIV_UDTRÆK;                                       
               RETURN;                                                  
            END;                                                        
                                                                        
         BH60101M = ''; /* INITIER HJÆLPEFELTER */                      
                                                                        
         CN570.C_SÆLGER   = '0';                                        
         CN570.C_2PCT     = '0';                                        
         CN570.C_2PCT_FOR = '0';                                        
         CN570.C_2PCT_ORD = '0';                                        
         C_2PCTBBR        = '0';                                        
         C_2PCTESR        = '0';                                        
                                                                        
         /*************************************************************/
         /* HEREFTER BLIVER OVENNÆVNTE 4 KODER EVT. OPSAT.            */
         /*                                                           */
         /* C_SÆLGER : 0 BETYDER PERSONEN ER IKKE EN SÆLGER           */
         /*          : 1 BETYDER PERSONEN ER EN SÆLGER                */
         /*                                                           */
         /* C_2PCT : 0 BETYDER OPFYLDER IKKE KRAV TIL 2% BEREGNING    */
         /*          1 KUN 1998 OPFYLDER KRAV MEN IKKE JVF. BBR       */
         /*          2 KUN 1998 OPFYLDER KRAV MEN KUN JVF. BBR        */
         /*          3 KUN 1998 OPFYLDER KRAV BÅDE JVF. BBR OG 8 ÅRS  */
         /*          4 KUN 1999 OPFYLDER KRAV MEN IKKE JVF. BBR       */
         /*          5 KUN 1999 OPFYLDER KRAV MEN KUN JVF. BBR        */
         /*          6 KUN 1999 OPFYLDER KRAV BÅDE JVF. BBR OG 8 ÅRS  */
         /*          7 1998+1999 OPFYLDER KRAV MEN IKKE JVF. BBR      */
         /*          8 1998+1999 OPFYLDER KRAV MEN KUN JVF. BBR       */
         /*          9 1998+1999 OPFYLDER KRAV JVF BBR OG/ELLER 8 ÅRS */
         /*                                                           */
         /* C_2PCT_ORD                                                */
         /* OG                                                        */
         /* C_2PCT_FOR : 0 BETYDER OPFYLDER IKKE EFTERFØLGENDE KRAV:  */
         /*            1 OPFYLDER MINDST 1 AF BETINGELSERNE 67 ÅR     */
         /*            2 OPFYLDER MINDST 1 AF BETINGELSERNE 60-66 ÅR  */
         /*            3 OPFYLDER MINDST 1 AF BETINGELSERNE < 60 ÅR   */
         /*            4 DISPENSATION OVERFØRT FRA ANDEN EJER (OPSÆT- */
         /*              TES EVT. I BH64400(SAMKØRING AF ALLE EJERE). */
         /*                                                           */
         /*************************************************************/
                                                                        
         CALL TEST_BBROPLYS;                                            
                                                                        
         /* UNDERSØG OM EJEREN ER */                                    
         /* ER PENSIONIST I 1998  */                                    
                                                                        
         HJ_CPR = CN570.DCPRCIR;                                        
         IF PERSON_ALDER_OK(BEREGN_ÅR1,HJ_CPR,'1')                      
         THEN                                                           
            HJ67 = '1';                                                 
         ELSE                                                           
            HJ67 = ' ';                                                 
                                                                        
         IF CN570.DCPRCIR = MTFIL.DCPRN & ^EOF_BH655                    
         THEN                                                           
            CALL PERSON_PÅ_MT;                                          
         ELSE                                                           
            CALL PERSON_EJ_PÅ_MT;                                       
                                                                        
         CALL OPSÆT_C_2PCT; /* OPFYLDES BBR OG/ELLER 8 ÅR'S REGEL */    
                                                                        
         CALL SKRIV_UDTRÆK;                                             
                                                                        
 END BEHANDL;                                                           
 /*********************************************************************/
 /*      OPSÆT KODE(R) FOR HVILKEN TYPE EJEREN ER JVF. EGEN ALDER     */
 /*      OG/ELLER EVT. LEJEVÆRDIKODE(3 ELLER 4).                      */
 /*********************************************************************/
                                                                        
 PERSON_EJ_PÅ_MT:                                                       
         PROC;                                                          
                                                                        
         TV.EJMT = TV.EJMT + 1;                                         
                                                                        
         /* KONTROLLER OM LEJEVÆRDIKODE 3 ELLER 4 OPFYLDES */           
                                                                        
         HJ_CLEJE = CN570.CLEJE;    /* CN570.CLEJE MÅ IKKE ÆNDRES HER */
                                    /* MEN FØRST I BH60100 (BEREGNING)*/
                                                                        
         IF HJ_CLEJE = '3' !                                            
            HJ_CLEJE = '4'                                              
         THEN                                                           
            DO;                                                         
               IF CN570.CCIVS = 'E' !                                   
                  CN570.CCIVS = 'L'                                     
               THEN;                                                    
               ELSE                                                     
                  HJ_CLEJE = ' ';                                       
            END;                                                        
                                                                        
         /* FØRST LAVES EVT. OPSÆTNING FOR AKTUELT INDKOMSTÅR */        
                                                                        
         IF HJ67     = '1' !                                            
            HJ_CLEJE = '3'                                              
         THEN                                                           
            C_2PCT_ORD = '1';                                           
         ELSE                                                           
            IF HJ_CLEJE = '4'                                           
            THEN                                                        
               C_2PCT_ORD = '2';                                        
                                                                        
         /* SÅ LAVES EVT. OPSÆTNING FOR AKTUELT FORSKUDSÅR */           
                                                                        
         /* UNDERSØG OM EJEREN ER */                                    
         /* ER PENSIONIST I 1999  */                                    
                                                                        
         HJ_CPR = CN570.DCPRCIR;                                        
         IF PERSON_ALDER_OK(BEREGN_ÅR2,HJ_CPR,'1')                      
         THEN                                                           
            HJ67 = '1';                                                 
         ELSE                                                           
            HJ67 = ' ';                                                 
                                                                        
         IF HJ67     = '1' !                                            
            HJ_CLEJE = '3'                                              
         THEN                                                           
            C_2PCT_FOR = '1';                                           
         ELSE                                                           
            IF HJ_CLEJE = '4'                                           
            THEN                                                        
               C_2PCT_FOR = '2';                                        
                                                                        
 END PERSON_EJ_PÅ_MT;                                                   
 /*********************************************************************/
 /*      OPSÆT KODE(R) FOR HVILKEN TYPE EJEREN ER JVF. EGEN ALDER     */
 /*      OG/ELLER EVT. ÆGTEFÆLLES ALDER SAMT STATUS IFLG. MT-FIL      */
 /*      OG/ELLER LEJEVÆRDIKODE I ESR(3 OG 4).                        */
 /*********************************************************************/
                                                                        
 PERSON_PÅ_MT:                                                          
         PROC;                                                          
                                                                        
         TV.FMT = TV.FMT + 1;                                           
                                                                        
         /* KONTROLLER OM LEJEVÆRDIKODE 3 ELLER 4 OPFYLDES */           
                                                                        
         HJ_CLEJE = CN570.CLEJE;    /* CN570.CLEJE MÅ IKKE ÆNDRES HER */
                                    /* MEN FØRST I BH60100 (BEREGNING)*/
                                                                        
         IF (MTFIL.CSBSK(INDKÅR) = '2' !                                
             MTFIL.CSBSK(INDKÅR) = '3' !                                
             MTFIL.CSBSK(INDKÅR) = '5' !                                
             MTFIL.CSBSK(INDKÅR) = '6') &                               
            (HJ_CLEJE = '3' ! HJ_CLEJE = '4')                           
         THEN                                                           
            DO;                                                         
               /* LEJEVÆRDIKODEN SLETTES, HVIS */                       
               /* PERSONEN HAR GIFTET SIG IGEN */                       
               HJ_CLEJE = ' ';                                          
            END;                                                        
                                                                        
         IF (HJ_CLEJE       <= '0' &                                    
             MTFIL.CDØDÆGTF >  '0' &                                    
             HJ67 ^= '1')                                               
         THEN                                                           
            DO;                                                         
               /* LEJEVÆRDIKODE 3 / 4 OPSÆTTES, */                      
               /* HVIS ÆGTEFÆLLEN ER DØD I 1998 */                      
               IF MTFIL.CDØDÆGTF = '1'                                  
               THEN                                                     
                  HJ_CLEJE = '3';        /* ENKE EFTER 67-ÅRIG */       
               ELSE                                                     
                  HJ_CLEJE = '4';       /* ENKE EFTER FØRTIDSPENS */    
            END;                                                        
                                                                        
         IF HJ_CLEJE = '3' !                                            
            HJ_CLEJE = '4'                                              
         THEN                                                           
            DO;                                                         
               IF MTFIL.CCIVS = 'E' !                                   
                  MTFIL.CCIVS = 'L'                                     
               THEN;                                                    
               ELSE                                                     
                  HJ_CLEJE = ' ';                                       
            END;                                                        
                                                                        
         /* FØRST LAVES EVT. OPSÆTNING FOR AKTUELT INDKOMSTÅR */        
                                                                        
         IF MTFIL.DHENVN(INDKÅR) > 0                                    
         THEN                                                           
            DO;                                                         
               /* UNDERSØG OM ÆGTEFÆLLE */                              
               /* ER PENSIONIST I 1998  */                              
               HJ_CPR = MTFIL.DHENVN(INDKÅR);                           
               IF PERSON_ALDER_OK(BEREGN_ÅR1,HJ_CPR,'1')                
               THEN                                                     
                  HJHNV67 = '1';                                        
               ELSE                                                     
                  HJHNV67 = ' ';                                        
            END;                                                        
                                                                        
         IF HJ67     = '1' !                                            
            HJHNV67  = '1' !                                            
            HJ_CLEJE = '3'                                              
         THEN

 /**********************************************************************
 *   UNDERSØG OM PERSONEN HAR EJET EJENDOMMEN I MINDST 8 ÅR            *
 *   PR. 31.12 I AKTUELT INDKOMSTÅR OG SAMTIDIG OPFYLDER EN            *
 *   AF BETINGELSERNE FOR HELÅRSBEBOELSE.                              *
 *   C_2PCT OPSÆTTES UDFRA 8 ÅRS REGLEN OG/ELLER DISP. JVF. BBR.       *
 **********************************************************************/
                                                                        
 OPSÆT_C_2PCT:                                                          
         PROC;                                                          
                                                                        
         /* FØRST UNDERSØGES AKTUELT INDKOMSTÅR */                      
                                                                        
         IF TEST_EJET_8ÅR_OK(CN570.DOVTG,HJ_DATO1) &                    
            CN570.C_2PCT_ORD > '0'                                      
         THEN                                                           
            C_2PCTESR = '1';                        /* 8 ÅR OK I 1998 */
         ELSE                                                           
            C_2PCTESR = '0';                        /* 8 ÅR OK I 1998 */
                                                                        
         /* SÅ UNDERSØGES AKTUELT FORSKUDSÅR */                         
                                                                        
         IF TEST_EJET_8ÅR_OK(CN570.DOVTG,HJ_DATO2) &                    
            CN570.C_2PCT_FOR > '0'                                      
         THEN                                                           
            DO;                                                         
              IF C_2PCTESR = '0'                                        
              THEN                                                      
                 C_2PCTESR = '2';             /* 8 ÅR OK I 1999 */      
              ELSE                                                      
                 C_2PCTESR = '3';             /* 8 ÅR OK I 1998+1999*/  
            END;                                                        
                                                                        
         /* SÅ OPSÆTTES CN570.C_2PCT        */                          
         /* P.G.A. AF OM BBR OG/ELLER 8 ÅRS */                          
         /* REGEL GIVER DISPENSATION.       */                          
                                                                        
         SELECT;                                                        
           WHEN (C_2PCTBBR = '0' & C_2PCTESR = '1')                     
             DO;                                                        
               CN570.C_2PCT= '1';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '0' & C_2PCTESR = '2')                     
             DO;                                                        
               CN570.C_2PCT= '4';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '0' & C_2PCTESR = '3')                     
             DO;                                                        
               CN570.C_2PCT= '7';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '1' & C_2PCTESR = '0')                     
             DO;                                                        
               CN570.C_2PCT= '2';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '1' & C_2PCTESR = '1')                     
             DO;                                                        
               CN570.C_2PCT= '3';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '1' & C_2PCTESR = '2')                     
             DO;                                                        
               CN570.C_2PCT= '9';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '1' & C_2PCTESR = '3')                     
             DO;                                                        
               CN570.C_2PCT= '9';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '2' & C_2PCTESR = '0')                     
             DO;                                                        
               CN570.C_2PCT= '5';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '2' & C_2PCTESR = '1')                     
             DO;                                                        
               CN570.C_2PCT= '9';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '2' & C_2PCTESR = '2')                     
             DO;                                                        
               CN570.C_2PCT= '6';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '2' & C_2PCTESR = '3')                     
             DO;                                                        
               CN570.C_2PCT= '9';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '3' & C_2PCTESR = '0')                     
             DO;                                                        
               CN570.C_2PCT= '8';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '3' & C_2PCTESR = '1')                     
             DO;                                                        
               CN570.C_2PCT= '9';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '3' & C_2PCTESR = '2')                     
             DO;                                                        
               CN570.C_2PCT= '9';                                       
             END;                                                       
           WHEN (C_2PCTBBR = '3' & C_2PCTESR = '3')                     
             DO;                                                        
               CN570.C_2PCT= '9';                                       
             END;                                                       
           OTHERWISE                                                    
               CN570.C_2PCT= '0';                                       
         END;                                                           
                                                                        
         IF CN570.C_2PCT > '0' &                                        
           (CN570.DAFSTÅ >  ÅR0101 &   /* HVIS EJEREN HAR SOLGT     */  
            CN570.DAFSTÅ <= ÅR1231)    /* EJENDOMMEN I INDKOMSTÅRET */  
         THEN                                                           
            CN570.C_SÆLGER = '1';                                       
         ELSE                                                           
            CN570.C_SÆLGER = '0';                                       
                                                                        
 END OPSÆT_C_2PCT;
