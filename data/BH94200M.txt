 TEST_EJET_ANTAL_ÅR_OK: PROC (HJ_STARTDATO, HJ_SLUTDATO,                
                              HJ_ANTAL_ÅR ) RETURNS (BIT(1));           
                                                                        
 DCL     RETURBIT        BIT       (1);                                 
 DCL     HJ_STARTDATO    FIXED      (9);                                
 DCL     HJ_SLUTDATO     FIXED      (9);                                
 DCL     HJ_ANTAL_ÅR     BIN FIXED (15);                                
 DCL     DATO_DIF        FIXED      (9);                                
                                                                        
         RETURBIT = '0'B;                                               
                                                                        
         IF HJ_STARTDATO = 0                                            
         THEN                                                           
            RETURBIT = '1'B;                                            
         ELSE                                                           
            DO;                                                         
               DATO_DIF = HJ_SLUTDATO - HJ_STARTDATO;                   
               IF DATO_DIF >= (10000 * HJ_ANTAL_ÅR)                     
               THEN                                                     
                  RETURBIT = '1'B;                                      
            END;                                                        
                                                                        
         RETURN (RETURBIT);                                             
                                                                        
 END TEST_EJET_ANTAL_ÅR_OK;
