 BEHANDL: PROC;                                                         
                                                                        
         IF DCLBE77000T.EVSSEJ_SID_CSLET < '1' &                        
            DCLBE78000T.TILEVS_CENKE ^= '5'                             
            /* ejendom/ejerskab er IKKE slettet      */                 
            /* og CENKE er ikke allerede opsat til 5 */                 
         THEN DO;                                                       
            /* opdater og skriv, læs næste 944 */                       
            DCLBE78000T.TILEVS_CENKE = '5';                             
            IF BE55600M.AFVIKLINGS_ID = 'OPRET_EVS_SLUT'                
            THEN DO;                                                    
               DCLBE78000T.TILFORSK_COPR_CENKE = 'S';                   
            END;                                                        
            ELSE DO;                                                    
               DCLBE78000T.TILFORSK_COPR_CENKE = 'F';                   
            END;                                                        
            /* Justering af persontype felt 756/758  */                 
            /* Svarer til ALDERSRET_SLUT=JA i BE94300L*/                
            IF ER_BEGRÆNSET_SKATTEPLIGTIG                               
            THEN DO;                                                    
               IF DCLBE78000T.EKOMNR = 999                              
               THEN DO;                                                 
                  IF HJBEGRÆNSET_TP = '1' !                             
                     HJBEGRÆNSET_TP = '3'                               
                  THEN DO;                                              
                      DCLBE78000T.TILEVS_CPTYPSLUT = '1';               
                      DCLBE78000T.TILEVS_CPTYPFOR = '1';                
                  END;                                                  
                  ELSE DO;  /* type 8 - domicileret */                  
                     DCLBE78000T.TILEVS_CPTYPSLUT = '4';                
                     DCLBE78000T.TILEVS_CPTYPFOR = '4';                 
                  END;                                                  
               END;                                                     
               ELSE DO;                                                 
                  DCLBE78000T.TILEVS_CPTYPSLUT = '4';                   
                  DCLBE78000T.TILEVS_CPTYPFOR  = '4';                   
               END;                                                     
            END;                                                        
            ELSE DO;                                                    
               DCLBE78000T.TILEVS_CPTYPSLUT = '3';                      
               DCLBE78000T.TILEVS_CPTYPFOR  = '3';                      
            END;                                                        
            CALL UPDATE_RESULT;                                         
         END;                                                           
         ELSE DO;                                                       
            /* ejendommen er slettet, så der overføres */               
            /* uden opdatering eller CENKE er allerede */               
            /* sat til 5 */                                             
            ANTAL_EJ_OPDATERET = ANTAL_EJ_OPDATERET + 1;                
            /* CALL UPDATE_RESULT; QEB */                               
         END;                                                           
                                                                        
 END BEHANDL;

 /* UNDERSØG OM EN BORGER ER BEGRÆNSET SKATTEPLIGTIG. FINDES MT DATA */

 /********************************************************************/ 
 ER_BEGRÆNSET_SKATTEPLIGTIG:PROC RETURNS(BIT(1));                       
                                                                        
   DCL HJBEGRÆNSET_BIT BIT(1);                                          
                                                                        
   HJBEGRÆNSET_BIT = NEJ;                                               
   HJBEGRÆNSET_TP = '0';                                                
                                                                        
   IF DCLBE76000T.HIST_OPD_DETTE_CSKPLOM >= '0'                         
   THEN                                                                 
     DO;                                                                
       IF DCLBE76000T.HIST_OPD_DETTE_CSKPLOM = '1' !                    
          DCLBE76000T.HIST_OPD_DETTE_CSKPLOM = '3' !                    
          DCLBE76000T.HIST_OPD_DETTE_CSKPLOM = '8'                      
       THEN                                                             
         DO;                                                            
           HJBEGRÆNSET_BIT = JA;                                        
           HJBEGRÆNSET_TP = DCLBE76000T.HIST_OPD_DETTE_CSKPLOM;         
         END;                                                           
     END;                                                               
   ELSE                                                                 
     DO;                                                                
       IF DCLBE77000T.EVSSEJ_SID_CSKPLOMF = '1' !                       
          DCLBE77000T.EVSSEJ_SID_CSKPLOMF = '3' !                       
          DCLBE77000T.EVSSEJ_SID_CSKPLOMF = '8'                         
       THEN                                                             
         DO;                                                            
           HJBEGRÆNSET_BIT = JA;                                        
           HJBEGRÆNSET_TP =                                             
                DCLBE77000T.EVSSEJ_SID_CSKPLOMF;                        
         END;                                                           
     END;                                                               
                                                                        
   RETURN(HJBEGRÆNSET_BIT);                                             
                                                                        
 END ER_BEGRÆNSET_SKATTEPLIGTIG;

 UPDATE_RESULT: PROC;                                                   
                                                                        
         DCLBE78000T.PROGRAMNAVN    = WS_PROGRAMNAVN;                   
         DCLBE78000T.DB2FUNKTION    = 'U';

             PROGRAMNAVN          = :WS_PROGRAMNAVN,                    
             DB2FUNKTION          = 'U',                                
             TILEVS_CENKE         = :DCLBE78000T.TILEVS_CENKE,          
             TILFORSK_COPR_CENKE  = :DCLBE78000T.TILFORSK_COPR_CENKE,   
             TILEVS_CPTYPSLUT     = :DCLBE78000T.TILEVS_CPTYPSLUT,      
             TILEVS_CPTYPFOR      = :DCLBE78000T.TILEVS_CPTYPFOR        
                                                                        
            WHERE TIL_EVS_AAR   = :WS_TIL_EVS_AAR                       
            AND   AFVIKLINGS_ID = :WS_FORSKUD_SLUT_TXT                  
            AND   DATA_ID       = :BE55600M.DATA_ID                     
            AND   DCPRN         = :WS_DCPRN                             
            AND   EKOMNR        = :DCLBE78000T.EKOMNR                   
            AND   EEJDNR        = :DCLBE78000T.EEJDNR                   
            AND   EENHEDLBNR    = :DCLBE78000T.EENHEDLBNR               
            AND   EGENNR        = :DCLBE78000T.EGENNR;                  
                                                                        
         SELECT(SQLCA.SQLCODE);                                         
            WHEN(0) DO;                                                 
               ANT_SKR = ANT_SKR + 1;                                   
               ANT_UPD = ANT_UPD + 1;                                   
            END;                                                        
            OTHERWISE DO;                                               
               ABENDTXT = '101 UPDATE RESULT';                          
               CALL SQL_FEJL;                                           
            END;                                                        
         END;                                                           
                                                                        
 END UPDATE_RESULT;

         /* nulstil areal ?*/                                           
                                                                        
         EXEC SQL FETCH DETAILCURS                                      
                                                                        
         INTO :DCLBE76000T.HIST_OPD_DETTE_CSKPLOM,                      
              :DCLBE77000T.EVSSEJ_SID_CSKPLOMF,                         
              :DCLBE77000T.EVSSEJ_SID_CSLET,                            
              :DCLBE78000T.DCPRN,                                       
              :DCLBE78000T.TILEVS_CENKE,                                
              :DCLBE78000T.TILFORSK_COPR_CENKE,                         
              :DCLBE78000T.EKOMNR,                                      
              :DCLBE78000T.EEJDNR,                                      
              :DCLBE78000T.EENHEDLBNR,                                  
              :DCLBE78000T.EGENNR,                                      
              :DCLBE78000T.TILEVS_CPTYPSLUT,                            
              :DCLBE78000T.TILEVS_CPTYPFOR                              
         ;                                                              
                                                                        
         SELECT (SQLCA.SQLCODE);                                        
                                                                        
           WHEN (0);                                                    
                                                                        
           WHEN (100);                                                  
                                                                        
           OTHERWISE                                                    
             DO;                                                        
               ABENDTXT = '101 FETCH DETAILCURS';                       
               CALL SQL_FEJL;                                           
             END;                                                       
                                                                        
         END; /* SELECT*/                                               
                                                                        
 END FETCH_DETAILCURS;
