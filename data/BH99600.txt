 BH99600: /* OPRET MINI MTFIL(FRA FORSKUD) P.G.A. SIDSTE ÅRS MTFIL */   
      PROC OPTIONS (MAIN) REORDER;                                      
          /* OPRET 'NY' MINI MTFIL TIL TEST - HUSK EVT. STRUKTUR-ÆNDR */
 DCL     BR042I          FILE RECORD INPUT;                             
 DCL     BR142O          FILE RECORD OUTPUT;                            
 DCL     1 BR042,                                                       
           %INCLUDE BR04250N;                                           
 DCL     1 BR04259       BASED(ADDR(BR042)),                            
           %INCLUDE BR04259N;                                           
 DCL     1 BR142,                                                       
           %INCLUDE BR14250N;                                           
 DCL     1 BR14259       BASED(ADDR(BR142)),                            
           %INCLUDE BR14259N;                                           
 DCL     EOF_GLMT        BIT (1) INIT ('0'B);                           
 DCL     ADDR            BUILTIN;                                       
 DCL     ZS61900         ENTRY;                                         
         DEFAULT RANGE (*) STATIC;                                      
         CALL ZS61900;                                                  
         OPEN FILE (BR042I) TITLE ('BR04250I');                         
         OPEN FILE (BR142O) TITLE ('BR14250O');                         
         ON ENDFILE (BR042I)                                            
            EOF_GLMT = '1'B;                                            
         READ FILE (BR042I) INTO (BR042);                               
         DO WHILE (^EOF_GLMT);                                          
            IF BR042.DCPRN < 9999999999                                 
            THEN                                                        
               DO;                                                      
                  BR142 = '';                                           
                  BR142 = BR042, BY NAME;                               
                  IF BR142.CRGN < '00'                                  
                  THEN                                                  
                     BR142.CRGN = '00';                                 
                  IF BR142.CCIVS = 'D'                                  
                  THEN;                                                 
                  ELSE                                                  
                     WRITE FILE (BR142O) FROM (BR142);                  
               END;                                                     
            ELSE                                                        
               DO;                                                      
                  BR14259 = BR04259, BY NAME;                           
                  WRITE FILE (BR142O) FROM (BR14259);                   
               END;                                                     
            READ FILE (BR042I) INTO (BR042);                            
         END;                                                           
         CLOSE FILE (BR042I);                                           
         CLOSE FILE (BR142O);                                           
 END BH99600;
