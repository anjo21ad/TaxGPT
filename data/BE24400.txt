 BEHANDLING: PROC;

      INDV_0132_OPTAELT.FANTAL_LAES = INDV_0132_OPTAELT.FANTAL_LAES + 1;
      I = 2;                                                            
      HJ_POS = 0;                                                       
      IND_POS = 0;                                                      
      AFBRYD = NEJ;                                                     
      BITTAB(*) = '0'B;                                                 
      INDV_0132.STAT(*) = '0'B;                                         
      SELECT (INDIVID.INDVNR);                                          
        WHEN('0001') /* HEADER */                                       
          DO;                                                           
            CALL DAN_HEADER_INDV0132;                                   
          END;                                                          
        WHEN('0132') /* Ejendomsdata */                                 
          DO;                                                           
             CALL BEHANDL_DATA;                                         
             IF AFBRYD = NEJ                                            
             THEN                                                       
               DO;                                                      
                   INDV_0132.LAENGDE = INDV_0132_LAENGDE;/*ÅR-RET*/     
                   INDV_0132.STAT(*) = BITTAB(*);                       
                   CALL DAN_INDIVID_TIL_DB2_LOAD;                       
                   UDDATA = ' ';                                        
                   INDV_0132 = '';                                      
                   /*INDV_0132_OPTAELT.FANTAL_SKRIV =                   
                                 INDV_0132_OPTAELT.FANTAL_SKRIV + 1*/   
             END;                                                       
          END;                                                          
        WHEN('0999') /* TRAILER */                                      
          DO;                                                           
             TRAILER_SW = JA;                                           
          END;                                                          
        OTHER  /* FEJL */                                               
          DO;                                                           
             PUT SKIP LIST('FEJL I 0132 INPUTDATA I RECORDNR: ',        
                           INDV_0132_OPTAELT.FANTAL_LAES);              
             ABENDTXT =                                                 
                '250 FEJL I 0132 INPUTDATA';                            
             CALL ABEND_MODUL('BE24400');                     /*ÅR-RET*/
          END;                                                          
      END; /*SELECT INDIVIDNR */                                        
      CALL LÆS_INPUTFIL;                                                
   END;

   DO WHILE(AFBRYD=NEJ & (HJ_POS < (TAB_SIZE+1)) &                      
            I < (MAX_ANT_FELTER+1));                                    
                                                                        
      HJ_FELTNR = PLACERING_BASED.FELTNR(I);                            
      FELTBIT = '0'B;

      IF PLACERING_BASED(I).FELTNR > MAX_ANT_FELTER THEN                
         ;                                                              
      ELSE DO;                                                          
        HJ_FELTNR = PLACERING_BASED(I).FELTNR;                          
        IF IND_POS < (TAB_SIZE + 1) THEN DO;                            
 /* SÆTTER AFBRYD OP TIL '1'B NÅR DER ER FEJL */                        
                                                                        
           AFBRYD = KONVERTER_OG_INDSAET_FELT(OUTPUT_TXT,HJ_FELTNR,     
                                      FELT_POS_START,FELT_POS_SLUT);    
                                                                        
           IF AFBRYD=JA THEN                                            
            PUT DATA(PLACERING_char(I),OUTPUT_TXT,HJ_FELTNR,            
                     FELT_POS_START,FELT_POS_SLUT);                     
                                                                        
           I = I + 1;                                                   
         END;                                                           
      END;                                                              
   END;                                                                 
 END BEHANDL_DATA;

 /* Konverterer fra char 08 til char(10) DB2 date                     */
 /*                                                                   */
 /*-------------------------------------------------------------------*/
 KONV_CH8_DB2DATE: PROC (CHAR8)                                         
                             RETURNS (CHAR (10));                       
 DCL     CHAR8      CHAR (08);                                          
 DCL     CHAR10     CHAR (10);                                          
                                                                        
         IF CHAR8 = ''                                                  
         THEN                                                           
           DO;                                                          
              CHAR10 = '';                                              
           END;                                                         
         ELSE                                                           
           DO;                                                          
              CHAR10 = SUBSTR(CHAR8,1,4) !! '-' !!                      
                       SUBSTR(CHAR8,5,2) !! '-' !!                      
                       SUBSTR(CHAR8,7,2);                               
           END;                                                         
         RETURN (CHAR10);                                               
                                                                        
 END KONV_CH8_DB2DATE;

 SKRIV_FEJL_TEXT: PROC(NR,FEJLTYPE,FELTTYPE,START_IDX,SLUT_IDX,FELT);   
    DECLARE NR        BINARY FIXED(15);                                 
    DECLARE FEJLTYPE  BINARY FIXED(15);                                 
    DECLARE FELTTYPE  BINARY FIXED(15);                                 
    DECLARE START_IDX BINARY FIXED(15);                                 
    DECLARE SLUT_IDX  BINARY FIXED(15);                                 
    DECLARE FELT      CHAR(50);                                         
                                                                        
    SELECT(FEJLTYPE);                                                   
      WHEN (1013) DO;                                                   
        PUT SKIP LIST(' BE24400: FOR MANGE KARAKTERER. FEJL: ' !! NR);  
        PUT SKIP LIST('       PNR        :' !! INDV_0132.PNR);          
        PUT SKIP LIST('       0132_REC_NR:' !! FANTAL_LAES);            
        PUT SKIP LIST('       INP_FELT_NR:' !! I);                      
        PUT SKIP LIST('          FELTTYPE:' !! FELTTYPE);               
        PUT SKIP LIST('         START_IDX:' !! START_IDX);              
        PUT SKIP LIST('          SLUT_IDX:' !! SLUT_IDX);               
        PUT SKIP LIST('            LÆNGDE:' !! (SLUT_IDX-START_IDX+1)); 
        PUT SKIP LIST('              DATA:' !! SUBSTR(OUTPUT_TXT,1,30));
        PUT SKIP LIST('         INPUT_POS:' !! IND_POS);                
        PUT SKIP LIST('TYPE_CHAR_DEF(FELTTYPE).LGD ' !!                 
                       TYPE_CHAR_DEF(FELTTYPE).LGD);                    
        PUT SKIP LIST('TYPE_CHAR_DEF(FELTTYPE).FELTTXT ' !!             
                             TYPE_CHAR_DEF(FELTTYPE).FELTTXT);          
      END;                                                              
      WHEN (1014) DO;                                                   
              PUT SKIP LIST(' BE24400: FEJL NR : ' !! NR);              
              PUT SKIP LIST('PNR         ' !! INDV_0132.PNR);           
              PUT SKIP LIST('0132_REC_NR ' !! FANTAL_LAES);             
              PUT SKIP LIST('FELTTYPE    ' !! FELTTYPE);                
              PUT SKIP LIST('          I ' !! I);                       
              PUT SKIP LIST('TALLET ER FOR STORT ' !! FELT);            
              PUT SKIP LIST('FELTTYPE            ' !!                   
                             FELTTYPE_BESKRIVELSE(FELTTYPE));           
              PUT SKIP LIST('TYPE_CHAR_DEF(FELTTYPE).LGD ' !!           
                             TYPE_CHAR_DEF(FELTTYPE).LGD);              
              PUT SKIP LIST('TYPE_CHAR_DEF(FELTTYPE).FELTTXT ' !!       
                             TYPE_CHAR_DEF(FELTTYPE).FELTTXT);          
      END;                                                              
      OTHERWISE   DO;                                                   
              PUT SKIP LIST('*********FEJL************* ' !! NR);       
              PUT SKIP LIST('PNR         ' !! INDV_0132.PNR);           
              PUT SKIP LIST('0132_REC_NR' !! FANTAL_LAES);              
              PUT SKIP LIST('FELTTYPE  ' !! FELTTYPE);                  
              PUT SKIP LIST('I ' !! I);                                 
              PUT SKIP LIST('UKENDT FELTTYPE ' !! FELTTYPE);            
              PUT SKIP LIST('TYPE_CHAR_DEF(FELTTYPE).LGD ' !!           
                             TYPE_CHAR_DEF(FELTTYPE).LGD);              
              PUT SKIP LIST('TYPE_CHAR_DEF(FELTTYPE).FELTTXT ' !!       
                             TYPE_CHAR_DEF(FELTTYPE).FELTTXT);          
      END;                                                              
    END;                                                                
                                                                        
 END SKRIV_FEJL_TEXT;
