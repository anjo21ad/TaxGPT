 BH55350:                                                               
       PROC (PARMPTR) REORDER;                                          
                                                                        
 DCL   COPYRIGHT  CHAR   (30) STATIC                                    
                  INIT ('Copyright KMD A/S 2021');                      
                                                                        
 /*-------------------------------------------------------------------- 
  Henter info fra åbne række på Afvikling                               
                                                                        
  Formål:     Henter den åbne række på Afvikling (BQ72100T) og          
              returnerer værdierne.                                     
                                                                        
  Input:      -                                                         
                                                                        
  Output:     Værdier fra den åbne række på Afvikling (BQ72100T)        
                                                                        
  Kald:       ZS67000 - DB2 logning                                     
                                                                        
  Programmør: Steen Pedersen / SEP - februar 2021                       
                                                                        
 ---------------------------------------------------------------------- 
  Rettelseslog:                                                         
                                                                        
 --------------------------------------------------------------------*/ 
                                                                        
 /*-------------------------------------------------------------------- 
       Standard include                                                 
 --------------------------------------------------------------------*/ 
       %INCLUDE ZM2ENVB;                                                
       %INCLUDE ZM2BATA;              // Standard-rutiner               
       %INCLUDE ZM2DCLA;                                                
                                                                        
       ZM2APPL    = 'BEVS';                                             
       ZM2PROGRAM = 'BH55350';                                          
                                                                        
 /*-------------------------------------------------------------------- 
       Declare af kommunikationsareal                                   
 --------------------------------------------------------------------*/ 
 DCL   PARMPTR              PTR;                                        
                                                                        
 DCL   1 PARM BASED (PARMPTR),                                          
         %INCLUDE BH55350N;                                             
 /*-------------------------------------------------------------------- 
       DB2                                                              
 --------------------------------------------------------------------*/ 
       EXEC SQL INCLUDE SQLCA;        // SQL Kommunikationsareal        
       EXEC SQL INCLUDE BQ72100T;     // Afviklingstabel                
                                                                        
 /*-------------------------------------------------------------------  
       ENTRY definitioner                                               
 -------------------------------------------------------------------*/  
 DCL   ZS67000       ENTRY;           // DB2 logning                    
                                                                        
 /*-------------------------------------------------------------------- 
       Declare af variable                                              
 --------------------------------------------------------------------*/ 
                                                                        
 /*-------------------------------------------------------------------- 
       Hovedprogram                                                     
 --------------------------------------------------------------------*/ 
 // Hent åben række på Afvikling (BQ72100T)                             
 CALL HENT_AFVIKLING;                                                   
                                                                        
 // Afslut progrsmmet                                                   
 RETURN;                                                                
                                                                        
 /*-------------------------------------------------------------------- 
       Hent åben række på Afvikling                                     
 --------------------------------------------------------------------*/ 
 HENT_AFVIKLING: PROC;                                                  
                                                                        
       DCLBQ72100T = '';                                                
                                                                        
       EXEC SQL                                                         
          SELECT                                                        
             TIL_EVS_AAR,                                               
             AFVIKLINGS_ID,                                             
             DATA_ID,                                                   
             OPRET_TMS,                                                 
             OPRET_AF,                                                  
             AFSLUT_TMS,                                                
             AFSLUT_AF                                                  
          INTO                                                          
             :DCLBQ72100T.TIL_EVS_AAR,                                  
             :DCLBQ72100T.AFVIKLINGS_ID,                                
             :DCLBQ72100T.DATA_ID,                                      
             :DCLBQ72100T.OPRET_TMS,                                    
             :DCLBQ72100T.OPRET_AF,                                     
             :DCLBQ72100T.AFSLUT_TMS,                                   
             :DCLBQ72100T.AFSLUT_AF                                     
          FROM BQ72100T                                                 
          WHERE AFSLUT_TMS = '9999-12-31-23.59.59.999999';              
                                                                        
       SELECT (SQLCA.SQLCODE);                                          
          WHEN (0) DO;                                                  
             // De fundne værdier returneres                            
             PARM.TIL_EVS_AAR   = DCLBQ72100T.TIL_EVS_AAR;              
             PARM.AFVIKLINGS_ID = DCLBQ72100T.AFVIKLINGS_ID;            
             PARM.DATA_ID       = DCLBQ72100T.DATA_ID;                  
             PARM.OPRET_TMS     = DCLBQ72100T.OPRET_TMS;                
             PARM.OPRET_AF      = DCLBQ72100T.OPRET_AF;                 
             PARM.AFSLUT_TMS    = DCLBQ72100T.AFSLUT_TMS;               
             PARM.AFSLUT_AF     = DCLBQ72100T.AFSLUT_AF;                
          END;                                                          
          WHEN (100) DO;                                                
             CALL SQL_FEJL ('001',                                      
                            'Der er ikke fundet en åben afvikling');    
          END;                                                          
          OTHER DO;                                                     
             CALL SQL_FEJL ('002',                                      
                            'Fejl ved opslag på BQ72100T (Afvikling)'); 
          END;                                                          
       END;                                                             
                                                                        
 END HENT_AFVIKLING;                                                    
                                                                        
 /*-------------------------------------------------------------------- 
       SQL Fejl                                                         
 --------------------------------------------------------------------*/ 
 SQL_FEJL: PROC (CC, TXT);                                              
 DCL   CC            CHAR (03);                                         
 DCL   TXT           CHAR (41);                                         
                                                                        
       PUT SKIP LIST ('*****************************************');     
       PUT SKIP LIST ('DER ABENDES MED SQLCODE: ' !! SQLCA.SQLCODE);    
       PUT SKIP LIST ('ABENDTXT:                ' !! TXT);              
       PUT SKIP LIST ('*****************************************');     
                                                                        
       CALL ZS67000(SQLCA);                                             
                                                                        
       CALL ZM2ABEND(CC, TXT);                                          
                                                                        
 END SQL_FEJL;                                                          
                                                                        
 END BH55350;
