 /********************************************************************/ 
 /*  BE90200M - BE19100                                              */ 
 /*  Dannes DUMMY_ID via NEXT VAL i hhv. BE90205S                    */ 
 /*                                                                  */ 
 /********************************************************************/ 
                                                                        
 /********************************************************************/ 
 DAN_DUMMY_ID: PROC();                                                  
                                                                        
         EXEC SQL                                                       
          SELECT NEXT VALUE FOR BE90205S                                
          INTO :WK_DUMMY_ID                                             
          FROM  SYSIBM.SYSDUMMY1                                        
         ;                                                              
                                                                        
         SELECT (SQLCA.SQLCODE);                                        
           WHEN (0)                                                     
            DO;                                                         
              DUMMY_ID_DANNET = 'J';                                    
            END;                                                        
           WHEN (100)                                                   
            DO;                                                         
              DUMMY_ID_DANNET = 'N';                                    
              PUT SKIP LIST                                             
              ('BE90200M - Fejl ved DAN_DUMMY_ID');                     
            END;                                                        
           OTHERWISE                                                    
             DO;                                                        
               FEJLTEKST = '306 DAN DUMMY_ID';                          
               CALL SQL_FEJL;                                           
             END;                                                       
         END; /* SELECT*/                                               
 END DAN_DUMMY_ID;
