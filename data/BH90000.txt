 /*          OPRET EVT. ÆGTEFÆLLE SOM 'NY' ESR-EJER NÅR DER ER FORE-  */

 /*                                                                   */
 /*          FUNKTIONEN UDFØRES KUN HVIS DEN SAMLEDE EJERANDEL JVF.   */
 /*          ESR OG GL. SLUTÅR STEMMER OVERENS SAMT  H V I S          */
 /*          ÆGTEFÆLLERNE ER SAMBESKATTET I BÅDE GL. OG NYT SLUTÅR.   */
 /*********************************************************************/
 OPSÆT_ÆGTF_FORDEL: PROC;                                               
                                                                        
 DCL ESR_SUM FIXED (5);                                                 
 DCL ESR_ANT FIXED (5);                                                 
 DCL EVS_SUM FIXED (5);                                                 
 DCL EVS_ANT FIXED (5);                                                 
                                                                        
         ESR_SUM = 0;                                                   
         ESR_ANT = 0;                                                   
         EVS_SUM = 0;                                                   
         EVS_ANT = 0;                                                   
                                                                        
         IF BH900.ESR.SORTCPR = 0                                       
         THEN                                                           
            RETURN;   /* INGEN EJERE JVF. ESR */                        
                                                                        
         /* KONTROLLER OM DE INVOLVEREDE ÆGTEFÆLLER ER              */  
         /* INDBERETTET MED KODE FOR ÆGTEFÆLLEFORDELING I GL.SLUTÅR */  
         /* DA DE ELLERS IKKE SKAL BEHANDLES HER.                   */  
                                                                        
         IF BH900.EVS_SLUT_EJER.SIDSTE.CÆGTOPR > '0' &                  
            BH900.EVS_SLUT_ÆGTF.SIDSTE.CÆGTOPR > '0'                    
         THEN;                                                          
         ELSE                                                           
            RETURN;                                                     
                                                                        
         IF (BH900.ESR.CSBSK = '2' !                                    
             BH900.ESR.CSBSK = '3' !                                    
             BH900.ESR.CSBSK = '5' !                                    
             BH900.ESR.CSBSK = '6') &                                   
            (BH900.EVS_UDTRÆK.CSBSK = '2' !                             
             BH900.EVS_UDTRÆK.CSBSK = '3' !                             
             BH900.EVS_UDTRÆK.CSBSK = '5' !                             
             BH900.EVS_UDTRÆK.CSBSK = '6')                              
         THEN;                                                          
         ELSE                                                           
            RETURN;   /* ÆGTEFÆLLERNE ER IKKE SAMBESKAT- */             
                      /* TET I BÅDE GL. OG NYT SLUTÅR    */             
                                                                        
         IF BH900.ESR.EJER.DCPRCIR > 0 &                                
           (BH900.ESR.EJER.DAFSTÅ > 0 &                                 
            BH900.ESR.EJER.DAFSTÅ <= BH900.ESR.EJER.DRGNÅRTIL)          
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF (BH900.ESR.EJER.DCPRCIR > 0 &                               
             BH900.ESR.ÆGTF.DCPRCIR > 0) &                              
            (BH900.ESR.EJER.DAFSTÅ > 0 &                                
             BH900.ESR.EJER.DAFSTÅ <= BH900.ESR.ÆGTF.DRGNÅRTIL)         
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH900.ESR.EJER.DCPRCIR > 0 &                                
            BH900.ESR.EJER.DOVTG > BH900.ESR.EJER.DRGNÅRFRA             
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF (BH900.ESR.EJER.DCPRCIR > 0 &                               
             BH900.ESR.ÆGTF.DCPRCIR > 0) &                              
            BH900.ESR.EJER.DOVTG > BH900.ESR.ÆGTF.DRGNÅRFRA             
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF (BH900.ESR.EJER.DCPRCIR > 0 &                               
             BH900.ESR.ÆGTF.DCPRCIR > 0) &                              
            (BH900.ESR.ÆGTF.DAFSTÅ > 0 &                                
             BH900.ESR.ÆGTF.DAFSTÅ <= BH900.ESR.EJER.DRGNÅRTIL)         
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH900.ESR.ÆGTF.DCPRCIR > 0 &                                
           (BH900.ESR.ÆGTF.DAFSTÅ > 0 &                                 
            BH900.ESR.ÆGTF.DAFSTÅ <= BH900.ESR.ÆGTF.DRGNÅRTIL)          
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF (BH900.ESR.EJER.DCPRCIR > 0 &                               
             BH900.ESR.ÆGTF.DCPRCIR > 0) &                              
            BH900.ESR.ÆGTF.DOVTG > BH900.ESR.EJER.DRGNÅRFRA             
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH900.ESR.ÆGTF.DCPRCIR > 0 &                                
            BH900.ESR.ÆGTF.DOVTG > BH900.ESR.ÆGTF.DRGNÅRFRA             
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH900.EVS_SLUT_EJER.DCPRN > 0 &                             
            BH900.EVS_SLUT_EJER.SIDSTE.CSLET < '1' &                    
            BH900.EVS_SLUT_EJER.SIDSTE.DAFSTAA > 0                      
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH900.EVS_SLUT_ÆGTF.DCPRN > 0 &                             
            BH900.EVS_SLUT_ÆGTF.SIDSTE.CSLET < '1' &                    
            BH900.EVS_SLUT_ÆGTF.SIDSTE.DAFSTAA > 0                      
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH900.EVS_SLUT_EJER.DCPRN > 0 &                             
            BH900.EVS_SLUT_EJER.SIDSTE.CSLET < '1' &                    
            BH900.EVS_SLUT_EJER.SIDSTE.DOVTG >                          
            BH900.EVS_SLUT_EJER.SIDSTE.DRGNÅRFRA                        
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH900.EVS_SLUT_ÆGTF.DCPRN > 0 &                             
            BH900.EVS_SLUT_ÆGTF.SIDSTE.CSLET < '1' &                    
            BH900.EVS_SLUT_ÆGTF.SIDSTE.DOVTG >                          
            BH900.EVS_SLUT_ÆGTF.SIDSTE.DRGNÅRFRA                        
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH900.EVS_SLUT_EJER.DCPRN > 0 &                             
            BH900.EVS_SLUT_EJER.SIDSTE.CSLET < '1' &                    
            //BH900.EVS_SLUT_EJER.SIDSTE.CEJBEVS ^= '9' & US1549418     
            BH900.EVS_SLUT_EJER.SIDSTE.GPCTEJER_SW = 0                  
         THEN                                                           
            DO;                                                         
               EVS_ANT = EVS_ANT + 1;                                   
               EVS_SUM = EVS_SUM +                                      
               BH900.EVS_SLUT_EJER.SIDSTE.GPCTEJER * 100;               
            END;                                                        
                                                                        
         IF BH900.EVS_SLUT_ÆGTF.DCPRN > 0 &                             
            BH900.EVS_SLUT_ÆGTF.SIDSTE.CSLET < '1' &                    
            //BH900.EVS_SLUT_ÆGTF.SIDSTE.CEJBEVS ^= '9' &  US1549418    
            BH900.EVS_SLUT_ÆGTF.SIDSTE.GPCTEJER_SW = 0                  
         THEN                                                           
            DO;                                                         
               EVS_ANT = EVS_ANT + 1;                                   
               EVS_SUM = EVS_SUM +                                      
               BH900.EVS_SLUT_ÆGTF.SIDSTE.GPCTEJER * 100;               
            END;                                                        
                                                                        
         IF BH900.ESR.EJER.DCPRCIR > 0 &                                
            BH900.ESR.EJER.GANE ^= 0                                    
         THEN                                                           
            DO;                                                         
               ESR_SUM = ESR_SUM + BH900.ESR.EJER.GANE;                 
               ESR_ANT = ESR_ANT + 1;                                   
            END;                                                        
                                                                        
         IF BH900.ESR.ÆGTF.DCPRCIR > 0 &                                
            BH900.ESR.ÆGTF.GANE ^= 0                                    
         THEN                                                           
            DO;                                                         
               ESR_SUM = ESR_SUM + BH900.ESR.ÆGTF.GANE;                 
               ESR_ANT = ESR_ANT + 1;                                   
            END;                                                        
                                                                        
         IF ESR_SUM ^= EVS_SUM                                          
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF EVS_ANT = 0 !                                               
            EVS_SUM = 0                                                 
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF ESR_ANT = EVS_ANT &                                         
            ESR_ANT = 2                                                 
         THEN                                                           
            DO;                                                         
               IF BH900.EVS_SLUT_EJER.DCPRN = BH900.ESR.EJER.DCPRCIR &  
                  BH900.EVS_SLUT_ÆGTF.DCPRN = BH900.ESR.ÆGTF.DCPRCIR    
               THEN;                                                    
               ELSE                                                     
                  RETURN;                                               
               /* BEHANDL PERSONER SOM ER ÆGTEFÆLLER   */               
               /* BÅDE I GL. SLUTÅR SAMT I NYT SLUTÅR. */               
               IF (BH900.ESR.EJER.GANE =                                
                   BH900.EVS_SLUT_EJER.SIDSTE.GPCTEJER * 100) &         
                  (BH900.ESR.ÆGTF.GANE =                                
                   BH900.EVS_SLUT_ÆGTF.SIDSTE.GPCTEJER * 100)           
               THEN;                                                    
               ELSE                                                     
                  DO;                                                   
                     BH900.AKT_EJER.GANE =                              
                     BH900.EVS_SLUT_EJER.SIDSTE.GPCTEJER * 100;         
                     BH900.AKT_ÆGTF.GANE =                              
                     BH900.EVS_SLUT_ÆGTF.SIDSTE.GPCTEJER * 100;         
                     //BH900.TIL_EVS.CÆGTOPR = '4';  US1549418          
                     TV.ÆGTF_OPRET = TV.ÆGTF_OPRET + 1;                 
                  END;                                                  
               RETURN;                                                  
            END;                                                        
                                                                        
         /* KONTROLLER OM DE INVOLVEREDE PERSONER ER ÆGTEFÆLLER */      
         /* BÅDE I GL. SLUTÅR SAMT I NYT SLUTÅR.                */      
                                                                        
         IF (BH900.EVS_SLUT_EJER.DCPRN > 0 &                            
             BH900.EVS_SLUT_ÆGTF.DCPRN > 0 &                            
             BH900.ESR.EJER.DCPRCIR > 0 &                               
             BH900.EVS_SLUT_EJER.DCPRN = BH900.ESR.EJER.DCPRCIR) !      
            (BH900.EVS_SLUT_EJER.DCPRN > 0 &                            
             BH900.EVS_SLUT_ÆGTF.DCPRN > 0 &                            
             BH900.ESR.ÆGTF.DCPRCIR > 0 &                               
             BH900.EVS_SLUT_ÆGTF.DCPRN = BH900.ESR.ÆGTF.DCPRCIR)        
         THEN;                                                          
         ELSE                                                           
            RETURN;                                                     
                                                                        
         /* BEHANDL PERSONER HVOR DER ER FORETAGET */                   
         /* ÆGTEFÆLLEFORDELING I GL. SLUTÅR        */                   
                                                                        
         IF ESR_ANT = EVS_ANT &                                         
            ESR_ANT = 1                                                 
         THEN                                                           
            DO;             /* KUN EEN EJER BÅDE I ESR OG GL. SLUTÅR */ 
               CALL BEHANDL_KUN_EEN_EJER;                               
               RETURN;                                                  
            END;                                                        
                                                                        
         IF EVS_ANT = 1
