 /* Projekt:     EVS Forskud / EVS Slut                              */ 
 /* Programnavn: BH33800                                             */ 
 /* Programtype: Batch                                               */ 
 /* Funktion:    Forbrænder program, som valider lejlighedsnummer    */ 
 /*              for 2-familiehuse (ejendomstype 4 eller 5)          */ 
 /*              pr. indkomstår og EVS-kørsel (Slut eller Forskud).  */ 
 /*                                                                  */ 
 /*              De CN337I records, som har fået udfyldt TEKFEJLNR   */ 
 /*              af BH33700, bliver ikke valideret.                  */ 
 /*                                                                  */ 
 /*              Feltet TEKFEJLNR udfyldes med følgende fejlnumre:   */ 
 /*                 3800 - Antal enheder på aktuelt indkomstår,      */ 
 /*                        aktuel EVS-kørsel, kommunenummer og       */ 
 /*                        ejendomsnummer i EVS-enhedsanvendelses-   */ 
 /*                        tabellen er ikke 2 (ejendomstype 4) eller */ 
 /*                        1 (ejendomstype 5)                        */ 
 /*                 3801 - Antal enheder på aktuelt indkomstår +     */ 
 /*                        aktuel EVS-kørsel eller tidligere ind-    */ 
 /*                        komstår + tidligere EVS-kørsel Slut,      */ 
 /*                        kommunenummer og ejendomsnummer i         */ 
 /*                        EVS-enhedsanvendelsestabellen er ikke 2   */ 
 /*                        (ejendomstype 4) eller 1 (ejendomstype 5) */ 
 /*                 3802 - Antal enheder på aktuelt indkomstår,      */ 
 /*                        aktuel EVS-kørsel eller tidligere         */ 
 /*                        EVS-kørsel Forskud, kommunenummer og      */ 
 /*                        ejendomsnummer i                          */ 
 /*                        EVS-enhedsanvendelsestabellen er ikke 2   */ 
 /*                        (ejendomstype 4) eller 1 (ejendomstype 5) */ 
 /*                                                                  */ 
 /* Input:       BH337I   - Indhold lig CN570I                       */ 
 /*                         + felt EENHEDLBNR                        */ 
 /*                         + felt TEKFEJLNR                         */ 
 /*                         + felt ENHEDBELIG                        */ 
 /*                         + felt VEJR_ENHED                        */ 
 /*                         + felt HUSNR_ENHED                       */ 
 /*                         + felt ETAGE_ENHED                       */ 
 /*                         + felt SDØR_ENHED                        */ 
 /*                         (output fra BH33700)                     */ 
 /*                                                                  */ 
 /*              BH338P   - Parameter fil - Find EVS Kørsel type     */ 
 /*                                                                  */ 
 /*              DEBUGI   - Skal programmet debugge (put skip):      */ 
 /*                                                                  */ 
 /*                            Hvis  (filen DEBUGI ikke findes     ) */ 
 /*                            eller (DEBUG_FROM = 0 og DEBU_TO = 0) */ 
 /*                               Ingen debug                        */ 
 /*                            ellers                                */ 
 /*                               Debug alt for de valgte            */ 
 /*                               record numre i BH337I              */ 
 /*                                                                  */ 
 /*              BQ80300T - EVS-enhedsanvendelsestabel (DB2)         */ 
 /*                                                                  */ 
 /* I-O          BQ80300T - EVS-enhedssporbarhed (DB2)               */ 
 /*                                                                  */ 
 /* Output:      BH338O   - Indhold lig BH337I                       */ 
 /*                         + felt TEKFEJLNR                         */ 
 /*                                                                  */

 SELECT (BH338_PARM.FS_MRK);

     EVS_FORSKUD = JA;                                                  
     EVS_KOERSEL = EVS_KOERSEL_F;                                       
     AFVIKL_ID   = BH338_PARM.FORSKUD_TXT;

                  !! ' - ' !! AFVIKL_ID);                               
   END;

     EVS_SLUT    = JA;                                                  
     EVS_KOERSEL = EVS_KOERSEL_S;                                       
     AFVIKL_ID   = BH338_PARM.SLUT_TXT;

                  !! ' - ' !! AFVIKL_ID);

     EVS_KOERSEL = EVS_KOERSEL_U;

 /********************************************************************/ 
 /* Læs næste BH337I record                                          */

 READ_BH337I: PROC;                                                     
                                                                        
 DCL DISP_X           CHAR(9);                                          
 DCL DISP_9       BASED(ADDR(DISP_X))                                   
                      PIC 'Z.ZZZ.ZZ9';                                  
                                                                        
 DCL DISP_BH337I      CHAR(9);                                          
                                                                        
 PUTSKIP_THIS = NEJ;                                                    
                                                                        
 READ FILE(BH337I) INTO(BH337_AREA);                                    
                                                                        
 IF FLERE_BH337I THEN DO;                                               
   BH337I_READ = BH337I_READ + 1;                                       
                                                                        
   SELECT(BH337.TEKFEJLNR);                                             
     WHEN(0)   INGEN_FEJL = JA;                                         
     OTHERWISE INGEN_FEJL = NEJ;                                        
   END;                                                                 
                                                                        
   IF (DEBUG_REC.DEBUG_TO > 0                   )                       
    & (  BH337I_READ      > DEBUG_REC.DEBUG_FROM                        
       ! BH337I_READ      = DEBUG_REC.DEBUG_FROM)                       
    & (  BH337I_READ      < DEBUG_REC.DEBUG_TO                          
       ! BH337I_READ      = DEBUG_REC.DEBUG_TO  ) THEN DO;              
     PUTSKIP_THIS = JA;                                                 
   END;                                                                 
 END;                                                                   
                                                                        
 IF PUTSKIP_THIS THEN DO;                                               
   DISP_9      = BH337I_READ;                                           
   DISP_BH337I = TRIM(DISP_X,' ');                                      
                                                                        
   PUT SKIP LIST(DEBUG_TXT !! PROCNAME                                  
                 !! ' - record nr ' !! DISP_BH337I);                    
 END;                                                                   
                                                                        
 END READ_BH337I;                                                       
                                                                        
 /********************************************************************/ 
 /* Initier felterne EN_EJERBOLIG & TO_EJERBOLIG                     */ 
 /********************************************************************/ 
 FIND_EJERBOLIG_ANTAL: PROC;                                            
                                                                        
 %INCLUDE BH33701M;      /* Kode skal ligge her før koden i proc     */ 
                                                                        
 IF EN_EJERBOLIG THEN DO;                                               
   SELECT(BH337.TEKFEJLNR);                                             
     WHEN(0)   BH337I_OK_EN   = BH337I_OK_EN   + 1;                     
     OTHERWISE BH337I_FEJL_EN = BH337I_FEJL_EN + 1;                     
   END;                                                                 
 END;                                                                   
                                                                        
 IF TO_EJERBOLIG THEN DO;                                               
   SELECT(BH337.TEKFEJLNR);                                             
     WHEN(0)   BH337I_OK_TO   = BH337I_OK_TO   + 1;                     
     OTHERWISE BH337I_FEJL_TO = BH337I_FEJL_TO + 1;                     
   END;                                                                 
 END;                                                                   
                                                                        
 IF ^EN_EJERBOLIG                                                       
  & ^TO_EJERBOLIG THEN DO;                                              
   BH337I_REST = BH337I_REST + 1;                                       
 END;                                                                   
                                                                        
 END FIND_EJERBOLIG_ANTAL;                                              
                                                                        
 /********************************************************************/ 
 /* VT.1 - Såfremt antal enheder på aktuelt indkomstår, aktuel       */ 
 /*        EVS-kørsel, kommunenummer og ejendomsnummer i             */ 
 /*        EVS-enhedsanvendelsestabellen (BQ803) ikke er             */ 
 /*        2 (ejendomstype 4) eller 1 (ejendomstype 5),              */ 
 /*        så er det en fejl (TEKFEJLNR = 3800)                      */ 
 /********************************************************************/ 
 BEH_VT1: PROC;                                                         
                                                                        
 DCL CNT_ANTAL FIXED BIN(15) INIT(0);                                   
 DCL P1        FIXED BIN(15) INIT(0);                                   
 DCL P2        FIXED BIN(15) INIT(0);                                   
                                                                        
 IF EN_EJERBOLIG THEN DO;                                               
   P1 = 1;                                                              
   P2 = 1;                                                              
 END;                                                                   
 ELSE DO;                                                               
   P1 = 1;                                                              
   P2 = 2;                                                              
 END;                                                                   
                                                                        
 DCLBQ80300T                   = '';                                    
 DCLBQ80300T.TIL_EVS_AAR       = INDKAAR;                               
 DCLBQ80300T.AFVIKLINGS_ID     = AFVIKL_ID;                             
 DCLBQ80300T.EKOMNR            = BH337.EKOMNR;                          
 DCLBQ80300T.EEJDNR            = BH337.EEJDNR;                          
                                                                        
 EXEC SQL SELECT COUNT(*)                                               
          INTO  :CNT_ANTAL                                              
          FROM   BQ80300T                                               
          WHERE  TIL_EVS_AAR    = :DCLBQ80300T.TIL_EVS_AAR              
          AND    AFVIKLINGS_ID  = :DCLBQ80300T.AFVIKLINGS_ID            
          AND    EKOMNR         = :DCLBQ80300T.EKOMNR                   
          AND    EEJDNR         = :DCLBQ80300T.EEJDNR                   
          AND    EENHEDLBNR    IN(:P1,:P2);                             
                                                                        
 SELECT (SQLCA.SQLCODE);                                                
   WHEN(0) DO;                                                          
     VT1_CNT = VT1_CNT + 1;                                             
                                                                        
     IF EN_EJERBOLIG THEN DO;                                           
       SELECT(CNT_ANTAL);                                               
         WHEN(1) DO;                                                    
           VT1_CNT_OK_EN = VT1_CNT_OK_EN + 1;                           
         END;
