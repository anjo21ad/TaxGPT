  /********************************************************************/
 /*  MACRO: BQ24004M                                                 */ 
 /*      DECLARE AF ABEND-RUTINE                                     */ 
 /********************************************************************/ 
 DCL     ZS30101         ENTRY;              /* ABENDRUTINE          */ 
 DCL     ABENDKODE    FIXED       (2);                                  
 DCL     1 ABENDMEDD,                                                   
           2 ABENDMODUL CHAR      (7),                                  
           2 ABENDFILL1 CHAR      (1),                                  
           2 ABENDMEDNR CHAR      (2),                                  
           2 ABENDFILL2 CHAR      (1),                                  
           2 ABENDTXT   CHAR      (50);                                 
 DCL     ABENDRTKODE    FIXED     (2);                                  
 /********************************************************************/ 
 /*  MACRO: ABEND                                                    */ 
 /*  BRUGES AF PROGRAMMERNE                                          */ 
 /*  -  BE24000                                                      */ 
  /********************************************************************/
 ABEND_MODUL: PROC(PROGID);                                             
 DCL PROGID   CHAR (07);                                                
                                                                        
         ABENDKODE    = 5;                                              
         ABENDFILL1  = ' ';                                             
         ABENDFILL2  = ' ';                                             
         ABENDMODUL   = PROGID;                                         
         ABENDMEDNR   = '**';                                           
         ABENDRTKODE  = 2;                                              
         PUT SKIP LIST (ABENDTXT);                                      
         CALL ZS30101(ABENDKODE,ABENDMEDD,ABENDRTKODE);                 
 END  ABEND_MODUL;
