 /**********************************************************************
 *   BH64201M:                                                         *
 *   UNDERSØG OM PERSONEN HAR EJET EJENDOMMEN I MINDST 8 ÅR            *
 *   PR. 31.12 I ET GIVET INDKOMSTÅR.                                  *
 **********************************************************************/
                                                                        
 TEST_EJET_8ÅR_OK: PROC (HJ_DOVTG,HJ0101) RETURNS (BIT(1));             
                                                                        
 DCL     RETURBIT        BIT       (1);                                 
                                                                        
 DCL     HJ_DOVTG        FIXED     (9);                                 
 DCL     HJ0101          FIXED     (9);                                 
 DCL     ANTDAGE         FIXED     (9);                                 
                                                                        
         RETURBIT = '0'B;                                               
                                                                        
         IF HJ_DOVTG = 0                                                
         THEN                                                           
            RETURBIT = '1'B;                                            
         ELSE                                                           
            DO;                                                         
               ANTDAGE = HJ0101-HJ_DOVTG;                               
               IF ANTDAGE >= 80000                                      
               THEN                                                     
                  RETURBIT = '1'B;                                      
            END;                                                        
                                                                        
         RETURN (RETURBIT);                                             
                                                                        
 END TEST_EJET_8ÅR_OK;
