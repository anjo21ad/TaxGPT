               IF GEM_FORR_EJER_DAFSTÅ = 0 !                            
                  GEM_FORR_EJER_DAFSTÅ > GEM_FORR_EJER_DRGNÅRTIL        
               THEN                                                     
                  GEM_FORR_EJER_CTILKØB = 'J';

                   GEM_FORR_EJER_DAFSTÅ <=                              
                             GEM_FORR_EJER_DRGNÅRTIL) &

                  GEM_FORR_EJER_CTILKØB = ' ';

            GEM_FORR_CITYP = GEM_FORR_ÆGTE_CITYP;
