 /* OBS.:  UDFØRES HVIS DER ER BEREGNET LEJEVÆRDIER FOR EJENDOMMEN    */
 /*        OG PERSONEN IKKE ER UNDTAGET FOR BEREGNING                 */
 /*********************************************************************/
 PERSONLIG_EJD_LEJEVÆRDI:                                               
       PROC(TAB_LEJEV);                                                 
 DCL   1 TAB_LEJEV(3),                  /* EJENDOMMENS 3 LEJEVÆRDIER  */
         2 BLEJE      FIXED  (11),      /* LEJEV. NORMAL              */
         2 BLEJEFP    FIXED  (11),      /* LEJEV. FØRTIDSPENS./EFTERL.*/
         2 BLEJE67    FIXED  (11);      /* LEJEVÆRDI PENSIONIST       */
                                        /* 1=SIDSTE ÅRS VÆRDIER       */
                                        /* 2=NYT ÅRS VÆRDIER UREDUC.  */
                                        /* 3=NYT ÅRS VÆRDIER REDUC.   */
         /* PÅ HVILKET GRUNDLAG??? */                                   
         IF BH801.CPER_ORD = '3' /* EJER ELL. ÆGTEFÆLLE 67 */           
         THEN                                                           
            DO;                                                         
               IF BH801.PER_LEJEV(1).CREDTYP = 0 /* IKKE REDUCERET */   
               THEN                                                     
                  BH801.PER_LEJEV(1).BELØB =                            
                  TAB_LEJEV(2).BLEJE67;                                 
               ELSE                                                     
                  BH801.PER_LEJEV(1).BELØB =                            
                  TAB_LEJEV(3).BLEJE67;                                 
            END;                                                        
         IF BH801.CPER_ORD = '2' /* EJER ELL. ÆGTEFÆLLE FØRTIDSPENS */  
         THEN                                                           
            DO;                                                         
               IF BH801.PER_LEJEV(1).CREDTYP = 0 /* IKKE REDUCERET */   
               THEN                                                     
                  BH801.PER_LEJEV(1).BELØB =                            
                  TAB_LEJEV(2).BLEJEFP;                                 
               ELSE                                                     
                  BH801.PER_LEJEV(1).BELØB =                            
                  TAB_LEJEV(3).BLEJEFP;                                 
            END;                                                        
         IF BH801.CPER_ORD = '1' !                                      
            BH801.CPER_ORD = '4'                                        
         THEN                                                           
            DO;                                                         
               /* IKKE PENSIONIST ELL. FØRTIDSPENS */                   
               IF BH801.PER_LEJEV(1).CREDTYP = 0 /* IKKE REDUCERET */   
               THEN                                                     
                  BH801.PER_LEJEV(1).BELØB =                            
                  TAB_LEJEV(2).BLEJE;                                   
               ELSE                                                     
                  BH801.PER_LEJEV(1).BELØB =                            
                  TAB_LEJEV(3).BLEJE;                                   
            END;                                                        
 END PERSONLIG_EJD_LEJEVÆRDI;
