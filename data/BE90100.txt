 SKRIV_PERSON:PROC;                                                     
                                                                        
         FLAG_BOBEHA = ' ' ;                /* JXT:20-06-2023   */      
                                                                        
         IF BE901.SORT_KOMNR < 999 &                                    
            BE901.AKT_ÆGTF.DCPRCIR > 0 &                                
            BE901.AKT_ÆGTF.DCPRCIR = BE901.EVS_SLUT_ÆGTF.DCPRN &        
           (BE901.EVS_SLUT_ÆGTF.SIDSTE.BOBEHAKOD = 'B' !

            BE901.EVS_SLUT_ÆGTF.SIDSTE.CSLET < '1' &

          ((BE901.AKT_EJER.GANE + BE901.AKT_ÆGTF.GANE) <= 10000)

                       BE901.AKT_EJER.DCPRCIR = 0)

                   END;                                                 
                 OTHERWISE                                              
                   DO;                                                  
                      /* BÅDE AFDØDE OG EFTERLEVENDE ÆGTEFÆLLE */       
                      /* EJER EJENDOMMEN JVF. ESR              */       
                      BE901.AKT_EJER.GANE =                             
                      BE901.AKT_EJER.GANE + BE901.AKT_ÆGTF.GANE;

         /* denne med C dog checkes der ikke på ægtefælle men på ejer */

                        /* EJER EJENDOMMEN JVF. ESR                   */

                                           BE901.AKT_ÆGTF.GANE) = 10000 
                        THEN                                            
                           FLAG_BOBEHA = '4';                           
                        ELSE                                            
                           FLAG_BOBEHA = '3' ;                          
                     END;                                               
               END; /* SELECT */                                        
            END;                                                        
                                                                        
         /* Til Kontrol 5326: Ingen afgørelse vedrørende bo-behandling*/
         /* inden forskud 2024 (Sættes i BE85010)                     */
         CALL HENT_MANDTALSOPLYSNINGER;                                 
                                                                        
         IF  FLAG_BOBEHA             <= ' '                             
          &  BE901.SORT_KOMNR        < 999                              
          &  BE901.AKT_ÆGTF.DCPRCIR  > 0                                
          &  BE901.AKT_ÆGTF.DCPRCIR  = BE901.EVS_SLUT_ÆGTF.DCPRN

 /********************************************************************* 
 *         CHECK OM INDENLANDSK EJENDOM SKAL VIDEREFØRES              * 
 *         PÅ GRUNDLAG AF SLUT                                        * 
 *         UDVIDDET MED CEJBEVS = B OG CEJBEVS = C SAMT               * 
 *         CDATAFRA = 7 TIL FORSKUD 2020                              * 
 *********************************************************************/ 
 CHECK_OM_EJD_VIDERE: PROC;                                             
                                                                        
     DCL    INDENLANDSK_EJENDOM                         BIT (1);        
     DCL    PERSON_ER_GYLDIG                            BIT (1);        
     DCL    KODE_2PROM_NEDSLAG_STØRRE_LIG_10            BIT (1);        
                                                                        
     DCL    EJER_ER_DØD                                 BIT (1);        
     DCL    EJERSKAB_ER_SLETTET                         BIT (1);        
     DCL    EJERSKAB_ER_OPRETTET_VED_ÆGTEFÆLLEFORDELING BIT (1);        
     DCL    EJERSKAB_ER_SOLGT                           BIT (1);        
     DCL    EJERSKAB_ER_FRAFLYTTET                      BIT (1);        
     DCL    EJER_HAR_EN_EJERANDEL                       BIT (1);        
                                                                        
     DCL    GENERELLE_KRITERIER_ER_OPFYLDT              BIT (1);        
     DCL    REGEL1_KRITERIER_ER_OPFYLDT                 BIT (1);        
     DCL    REGEL2_KRITERIER_ER_OPFYLDT                 BIT (1);        
     DCL    REGEL3_KRITERIER_ER_OPFYLDT                 BIT (1);        
                                                                        
     /**************************************************/               
     /* G E N E R E L T  check for samtlige regler 1-3 */               
     /**************************************************/               
                                                                        
     /* Er ejendommen indenlandsk ? */                                  
     IF BE901.SORT_KOMNR < 999                                          
     THEN                                                               
       INDENLANDSK_EJENDOM = JA;                                        
     ELSE                                                               
       INDENLANDSK_EJENDOM = NEJ;                                       
                                                                        
     /* Er cprnr ok, og er persontypen sat i slutår ? */                
     IF BE901.AKT_EJER.DCPRCIR = 0 &                                    
        BE901.EVS_SLUT_EJER.DCPRN > 0 &                                 
        BE901.EVS_SLUT_EJER.SIDSTE.CPTYPSLUT > '0'                      
     THEN                                                               
       PERSON_ER_GYLDIG = JA;                                           
     ELSE                                                               
       PERSON_ER_GYLDIG = NEJ;                                          
                                                                        
     /* Er 2 promille nedslag større eller lig 10 ? */                  
     IF BE901.EVS_SLUT_EJER.SIDSTE.CNEDSL1 >= '10'                      
     THEN                                                               
       KODE_2PROM_NEDSLAG_STØRRE_LIG_10 = JA;                           
     ELSE                                                               
       KODE_2PROM_NEDSLAG_STØRRE_LIG_10 = NEJ;                          
                                                                        
     /* Er ejer død ? */                                                
     IF BE901.EVS_SLUT_EJER.SIDSTE.FEJLNR ^= '5004'                     
     THEN                                                               
       EJER_ER_DØD = NEJ;                                               
     ELSE                                                               
       EJER_ER_DØD = JA;                                                
                                                                        
     /* Er ejerskab slettet ? */                                        
     IF BE901.EVS_SLUT_EJER.SIDSTE.CSLET < '1'                          
     THEN                                                               
       EJERSKAB_ER_SLETTET = NEJ;                                       
     ELSE                                                               
       EJERSKAB_ER_SLETTET = JA;                                        
                                                                        
     /* Er ejerskab oprettet ved ægtefællefordeling ? */                
     IF BE901.EVS_SLUT_EJER.SIDSTE.CÆGTOPR < '1' &                      
        BE901.EVS_SLUT_EJER.SIDSTE.FEJLNR ^= '5027'                     
     THEN                                                               
       EJERSKAB_ER_OPRETTET_VED_ÆGTEFÆLLEFORDELING = NEJ;               
     ELSE                                                               
       EJERSKAB_ER_OPRETTET_VED_ÆGTEFÆLLEFORDELING = JA;                
                                                                        
     /* Er ejerskabet solgt ? */                                        
     IF BE901.EVS_SLUT_EJER.SIDSTE.DAFSTAA = 0                          
     THEN                                                               
       EJERSKAB_ER_SOLGT = NEJ;                                         
     ELSE                                                               
       EJERSKAB_ER_SOLGT = JA;                                          
                                                                        
     /* Er ejerskabet fraflyttet ? */                                   
     IF BE901.EVS_SLUT_EJER.SIDSTE.DFRAFLYT = 0                         
     THEN                                                               
       EJERSKAB_ER_FRAFLYTTET = NEJ;                                    
     ELSE                                                               
       EJERSKAB_ER_FRAFLYTTET = JA;                                     
                                                                        
     /* har ejerskabet en ejerandel ? */                                
     IF BE901.EVS_SLUT_EJER.SIDSTE.GPCTEJER_SW = 0 &                    
        BE901.EVS_SLUT_EJER.SIDSTE.GPCTEJER > 0                         
     THEN                                                               
       EJER_HAR_EN_EJERANDEL = JA;                                      
     ELSE                                                               
       EJER_HAR_EN_EJERANDEL = NEJ;                                     
                                                                        
     IF INDENLANDSK_EJENDOM = JA &                                      
        PERSON_ER_GYLDIG = JA &                                         
        EJER_ER_DØD = NEJ &                                             
        EJERSKAB_ER_SLETTET = NEJ &                                     
        EJERSKAB_ER_OPRETTET_VED_ÆGTEFÆLLEFORDELING = NEJ &             
        EJERSKAB_ER_SOLGT = NEJ &                                       
        EJERSKAB_ER_FRAFLYTTET = NEJ &                                  
        EJER_HAR_EN_EJERANDEL = JA &                                    
        KODE_2PROM_NEDSLAG_STØRRE_LIG_10 = JA                           
     THEN                                                               
       Generelle_kriterier_er_opfyldt = JA;                             
     ELSE                                                               
       Generelle_kriterier_er_opfyldt = NEJ;                            
                                                                        
                                                                        
     /**************************************************/               
     /* I N D I V I D U E L T  check for regel 1-3     */               
     /**************************************************/               
                                                                        
     /* REGEL 1 */                                                      
     IF BE901.EVS_SLUT_EJER.SIDSTE.CEJBEVS = '5' &                      
        BE901.EVS_SLUT_EJER.SIDSTE.CDATAFRA ^= '6'                      
     THEN                                                               
       Regel1_kriterier_er_opfyldt = JA;                                
     ELSE                                                               
       Regel1_kriterier_er_opfyldt = NEJ;                               
                                                                        
     /* REGEL 2 */                                                      
     IF(BE901.EVS_SLUT_EJER.SIDSTE.CEJBEVS < '8' !                      
        BE901.EVS_SLUT_EJER.SIDSTE.CEJBEVS = 'B' !                      
        BE901.EVS_SLUT_EJER.SIDSTE.CEJBEVS = 'C') &                     
       (BE901.EVS_SLUT_EJER.SIDSTE.CDATAFRA = '4' !                     
        BE901.EVS_SLUT_EJER.SIDSTE.CDATAFRA = '7')                      
     THEN                                                               
       Regel2_kriterier_er_opfyldt = JA;                                
     ELSE                                                               
       Regel2_kriterier_er_opfyldt = NEJ;                               
                                                                        
                                                                        
     /* REGEL 3 */
