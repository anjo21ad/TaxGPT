 SKAF_NYE_EJD_LEJEVÆRDIER:                                              
         PROC(TAB_LEJEV,NYE_LEJEV);                                     
                                                                        
 DCL   1 TAB_LEJEV(3),                /* EJENDOMMENS 3 LEJEVÆRDIER    */
         2 BLEJE      FIXED  (11),    /* LEJEV. NORMAL                */
         2 BLEJEFP    FIXED  (11),    /* LEJEV. FØRTIDSPENS./EFTERL.  */
         2 BLEJE67    FIXED  (11);    /* LEJEVÆRDI PENSIONIST         */
                                      /* 1=SIDSTE ÅRS VÆRDIER         */
                                      /* 2=NYT ÅRS VÆRDIER UREDUC.    */
                                      /* 3=NYT ÅRS VÆRDIER REDUC.     */
                                                                        
 DCL   1 NYE_LEJEV,                   /* EJENDOMMENS NYE LEJEVÆRDIER  */
         2 BLEJE      FIXED  (11),    /* LEJEV. NORMAL                */
         2 BLEJEFP    FIXED  (11),    /* LEJEV. FØRTIDSPENS./EFTERL.  */
         2 BLEJE67    FIXED  (11);    /* LEJEVÆRDI PENSIONIST         */
                                                                        
         IF TAB_LEJEV.BLEJE(3) > 0                                      
         THEN                                                           
            NYE_LEJEV.BLEJE = TAB_LEJEV.BLEJE(3);                       
         ELSE                                                           
            NYE_LEJEV.BLEJE = TAB_LEJEV.BLEJE(2);                       
                                                                        
         IF TAB_LEJEV.BLEJEFP(3) > 0                                    
         THEN                                                           
            NYE_LEJEV.BLEJEFP = TAB_LEJEV.BLEJEFP(3);                   
         ELSE                                                           
            NYE_LEJEV.BLEJEFP = TAB_LEJEV.BLEJEFP(2);                   
                                                                        
         IF TAB_LEJEV.BLEJE67(3) > 0                                    
         THEN                                                           
            NYE_LEJEV.BLEJE67 = TAB_LEJEV.BLEJE67(3);                   
         ELSE                                                           
            NYE_LEJEV.BLEJE67 = TAB_LEJEV.BLEJE67(2);                   
                                                                        
 END SKAF_NYE_EJD_LEJEVÆRDIER;
