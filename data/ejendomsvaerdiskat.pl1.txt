
/* PL/I PROGRAM: EJENDOMSVÆRDISKATTEBEREGNING */
EJENDOMSKATTEBEREGNING: PROCEDURE OPTIONS(MAIN);

   DECLARE EJENDOMSVÆRDI     FIXED DECIMAL(9,0);
   DECLARE FRITAGELSESGRÆNSE FIXED DECIMAL(9,0) INIT(300000);
   DECLARE SKATTEPROCENT     FIXED DECIMAL(5,2) INIT(1.00); /* 1% */
   DECLARE SKAT              FIXED DECIMAL(9,2);

   /* Indlæs ejendomsværdi */
   PUT SKIP LIST('Indtast ejendomsværdi:');
   GET LIST(EJENDOMSVÆRDI);

   /* Beregn ejendomsværdiskat */
   IF EJENDOMSVÆRDI <= FRITAGELSESGRÆNSE THEN
      SKAT = 0;
   ELSE
      SKAT = (EJENDOMSVÆRDI - FRITAGELSESGRÆNSE) * (SKATTEPROCENT / 100);

   /* Udskriv resultat */
   PUT SKIP LIST('Ejendomsværdiskat er: ', SKAT);

END EJENDOMSKATTEBEREGNING;
