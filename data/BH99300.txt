  PROCESS OPT(TIME);                                                    
 BH99300: /* UDTRÆK INDIVIDER TIL FEJLSØGNING */                        
         PROCEDURE OPTIONS (MAIN) REORDER;                              
 /*********************************************************************/
         DCL COPYRIGHT   CHAR      (30) STATIC                          
                         INIT ('COPYRIGHT KOMMUNEDATA A/S 2001');       
 /**********************************************************************
 *                                                                     *
 * INPUT  : B.H96000D  BEREGNINGSBÅND TIL EVS SLUT 2001                *
 *                                                                     *
 * OUTPUT : B.H99300D  UDTRÆK - FEJLNR. 5154                           *
 *          B.H99310D  UDTRÆK - DIVERSE PERSONER/EJENDOMME             *
 *          BH99301Y   KØRSELSKONTROLLISTE                             *
 *                                                                     *
 * PROGRAMMØR: AAGE RASMUSSEN  FEB 2002                                *
 *                                                                     *
 **********************************************************************/
         DEFAULT RANGE (*) STATIC UNALIGNED CONNECTED;                  
 /********************************************************************* 
 *                       DECLARE AF REGISTRE                          * 
 *********************************************************************/ 
 DCL     BH960I FILE RECORD INPUT;                                      
 DCL     BH9930O FILE RECORD OUTPUT;                                    
 DCL     BH9931O FILE RECORD OUTPUT;                                    
 /********************************************************************* 
 *       DECLARE AF UDTRÆK                                            * 
 *********************************************************************/ 
 DCL     BH960_AR        CHAR      (4000) VAR;                          
 DCL     1 BH960         BASED     (ADDR(BH960_AR)),                    
           2 LÆNGDE      BIN FIXED (15),                                
           %INCLUDE BH90000N;;                                          
 /********************************************************************* 
 *               DECLARE AF DIV. TÆLLEVÆRKER                          * 
 *********************************************************************/ 
 DCL     1 TV,                   /* TÆLLEVÆRKER TIL TOTALLISTEN */      
           2 SKRF5154    FIXED DEC (7),                                 
           2 SKRØVR      FIXED DEC (7);                                 
 /********************************************************************* 
 *       DECLARE AF BUILTIN'S                                         * 
 *********************************************************************/ 
 DCL     PLIDUMP         BUILTIN;                                       
 DCL     ADDR            BUILTIN;                                       
 /********************************************************************* 
 *       DECLARE AF ENTRY'S                                           * 
 *********************************************************************/ 
 DCL     ZS61900         ENTRY;                                         
 DCL     ZS38100         ENTRY OPTIONS (INTER ASM);   /* TIMESTAMP    */
 /*********************************************************************/
 /*      DECLARE AF PRINTLINIER                                       */
 /*********************************************************************/
 DCL    1 SIDELIN,                                                      
          2 CCA          CHAR      (01),                                
          2 FILLER       CHAR      (132)     INIT      (' ');           
 DCL     1 OVERLIN1,                                                    
           2 CCA         CHAR      (01),                                
           2 TEKST1      CHAR      (17)      INIT      (                
           'PROGRAM: BH99300L'),                                        
           2 F1          CHAR      (25)      INIT      (' '),           
           2 TEKST2      CHAR      (57)      INIT      (                
           'UDSØG TIL FEJLSØGNING VEDR. EVS SLUT 2001 '),               
           2 F2          CHAR      (12)      INIT      (' '),           
           2 TEKST3      CHAR      (11)      INIT      (                
           'UDSKREVET:'),                                               
           2 DATO        CHAR      (10);                                
 DCL     1 OVERLIN2,                                                    
           2 CCA         CHAR      (01),                                
           2 F1          CHAR      (46)      INIT      (' '),           
           2 TEKST1      CHAR      (86)      INIT                       
           ('          KØRSELSKONTROLLISTE');                           
 DCL     1 DETLIN1,                                                     
           2 CCA         CHAR      (01),                                
           2 TEKST       CHAR      (60),                                
           2 ANTAL       PIC       'ZZZ.ZZZ.ZZZ',                       
           2 F2          CHAR      (61)      INIT      (' ');           
         % INCLUDE ZZ20301M;                                            
 /********************************************************************* 
 *               DECLARE AF DIV. HJÆLPEFELTER                         * 
 *********************************************************************/ 
 DCL     PLIXOPT         CHAR      (40)      VAR STATIC                 
 EXTERNAL INIT ('NOTEST(ALL,,;)');                                      
 DCL     EOF_BH960       BIT       (1);                                 
 DCL     JA              BIT       (1)     INIT ('1'B);                 
 DCL     NEJ             BIT       (1)     INIT ('0'B);                 
                                                                        
 DCL     HJCPR           PIC       '(10)9';                             
                                                                        
 DCL     1 HJCPR1        BASED(ADDR(HJCPR)),                            
           2 F1          CHAR      (4),                                 
           2 FØDÅR       PIC       '99';                                
                                                                        
 DCL     UDDATO          CHAR      (10);                                
                                                                        
 DCL     DATO1           CHAR      (26);                                
                                                                        
 DCL     1 DATO2         DEF DATO1,                                     
           2 ÅR          CHAR      (04),                                
           2 F1          CHAR      (01),                                
           2 MD          CHAR      (02),                                
           2 F2          CHAR      (01),                                
           2 DG          CHAR      (02);                                
                                                                        
 /********************************************************************* 
 *       HOUSEKEEPING                                                 * 
 *********************************************************************/ 
                                                                        
         CALL ZS61900;                         /* UDSKRIVNING AF SSI */ 
                                                                        
         EOF_BH960 = NEJ;                                               
                                                                        
         ON ERROR                                                       
            CALL PLIDUMP ('SNHOB');                                     
                                                                        
         OPEN FILE (BH960I)  TITLE ('BH96000I');                        
         OPEN FILE (BH9930O) TITLE ('BH99300O');                        
         OPEN FILE (BH9931O) TITLE ('BH99310O');                        
                                                                        
         ON ENDFILE (BH960I)                                            
            EOF_BH960 = JA;                                             
                                                                        
         TV = '';                                                       
                                                                        
         CALL ZZ20390 ('*OPEN','BH99301Y');                             
                                                                        
         CALL ZS38100(DATO1);                                           
         UDDATO = DATO2.DG !! '/' !! DATO2.MD !!                        
                  '-' !! DATO2.ÅR;                                      
         OVERLIN1.DATO  = UDDATO;                                       
                                                                        
         SIDELIN.CCA  = ZZIK1;                                          
         OVERLIN1.CCA = ZZWS2;                                          
         OVERLIN2.CCA = ZZWS3;                                          
                                                                        
 /********************************************************************* 
 *       HOVEDSTYRING                                                 * 
 *********************************************************************/ 
                                                                        
         READ FILE (BH960I) INTO (BH960_AR);                            
                                                                        
         DO WHILE ( EOF_BH960 = NEJ );                                  
                                                                        
            CALL BEHANDL;                                               
                                                                        
            READ FILE (BH960I) INTO (BH960_AR);                         
                                                                        
         END;                                                           
                                                                        
         CALL AFSLUT;                                                   
 /********************************************************************* 
 *       PERSONER SOM EVT. SKAL UDTRÆKKES                             * 
 *********************************************************************/ 
 BEHANDL:                                                               
 PROC;                                                                  
                                                                        
         IF BH960.SORT_CPR = 99999999999     /* OPTÆLL.INDV. */         
         THEN                                                           
            RETURN;                                                     
                                                                        
         IF BH960.FEJLNR = 5154                                         
         THEN                                                           
            DO;                                                         
               TV.SKRF5154 = TV.SKRF5154 + 1;                           
               WRITE FILE (BH9930O) FROM (BH960_AR);                    
               RETURN;                                                  
            END;                                                        
                                                                        
         IF ((BH960.FEJLNR = 5022 !                                     
              BH960.FEJLNR = 5027) &                                    
              BH960.ESR.EKOMNR = 0) !                                   
            BH960.CUDLAND > '3'                                         
         THEN                                                           
            DO;                                                         
               TV.SKRØVR = TV.SKRØVR + 1;                               
               WRITE FILE (BH9931O) FROM (BH960_AR);                    
            END;                                                        
                                                                        
 END BEHANDL;                                                           
 /********************************************************************* 
 *       TOTALER UDSKRIVES.                                           * 
 *********************************************************************/ 
 TOTALLISTE:                                                            
 PROC;                                                                  
                                                                        
         CALL ZZ20300(SIDELIN,'01');                                    
         CALL ZZ20300(OVERLIN1,'01');                                   
         CALL ZZ20300(OVERLIN2,'01');                                   
                                                                        
         DETLIN1.CCA = ZZWS2;                                           
         DETLIN1.ANTAL = TV.SKRF5154;                                   
         DETLIN1.TEKST = 'ANTAL SKREVET FEJLNR. 5154        B.H99300D'; 
         CALL ZZ20300 (DETLIN1,'01');                                   
                                                                        
         DETLIN1.CCA = ZZWS2;                                           
         DETLIN1.ANTAL = TV.SKRØVR;                                     
         DETLIN1.TEKST = 'ANTAL ØVRIGE                      B.H99310D'; 
         CALL ZZ20300 (DETLIN1,'01');                                   
                                                                        
  END TOTALLISTE;                                                       
 /********************************************************************* 
 *       AFSLUT PROGRAM                                               * 
 *********************************************************************/ 
 AFSLUT:                                                                
 PROC;                                                                  
                                                                        
         CALL TOTALLISTE;                                               
                                                                        
         CALL ZZ20300((133)' ','99');                                   
                                                                        
         CLOSE FILE (BH960I),                                           
               FILE (BH9930O),                                          
               FILE (BH9931O);                                          
                                                                        
 END AFSLUT;                                                            
                                                                        
 END  BH99300;
