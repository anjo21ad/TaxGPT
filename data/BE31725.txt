 /*********************************************************************/
 /*      DECLARE AF ENTRIES OG BUILTINS                               */
 /*********************************************************************/
 DCL     ZS61900   ENTRY OPTIONS (INTER ASM);          /* UDSKRIV SSI */
 DCL     ZS38100   ENTRY OPTIONS (INTER ASM);          /* TIMESTAMP   */
 DCL     ADDR      BUILTIN;                                             
 /*********************************************************************/
 /*      DECLARE AF FILER                                             */
 /*********************************************************************/
 DCL     BE316I          FILE RECORD INPUT;                             
 DCL     BE317O          FILE RECORD OUTPUT;                            
 /*********************************************************************/
 /*      DECLARE AF INPUT                                             */
 /*********************************************************************/
 DCL     BE316_AR   CHAR      (1500) VAR;                               
                                                                        
 DCL     1 BE300              BASED(ADDR(BE316_AR)),                    
   4 LÆNGDE       BIN FIXED (15),                                       
   %INCLUDE BE30000N;;                                                  
                                                                        
 DCL     1 BE301              BASED(ADDR(BE316_AR)),                    
   4 LÆNGDE       BIN FIXED (15),                                       
   %INCLUDE BE30100N;;                                                  
                                                                        
 DCL     1 BE302              BASED(ADDR(BE316_AR)),                    
   3 LÆNGDE       BIN FIXED (15),                                       
   %INCLUDE BE30200N;;                                                  
                                                                        
 DCL     1 BE315              BASED(ADDR(BE316_AR)),                    
   3 LÆNGDE       BIN FIXED (15),                                       
   %INCLUDE BE31500N;;                                                  
                                                                        
 DCL     1 BE316              BASED(ADDR(BE316_AR)),                    
   2 LÆNGDE       BIN FIXED (15),                                       
   2 DCPRN        FIXED (11),          /* PERSONNUMMER                */
   2 EKOMNR       BIN FIXED (15) UNAL, /* KOMMUNENR                   */
   2 EEJDNR       BIN FIXED (31) UNAL, /* EJENDOMSNUMMER              */
   2 EENHEDLBNR   BIN FIXED (15) UNAL, /* ENHEDSNR                    */
   2 CFRATYPE     CHAR (1);                                             
                                                                        
 DCL     GEM_BE311_AR   CHAR      (50) VAR;                             
                                                                        
 DCL     1 GEM_BE311 BASED(ADDR(GEM_BE311_AR)),                         
           4 LÆNGDE       BIN FIXED (15),                               
           %INCLUDE BE31100N;;                                          
                                                                        
 /*********************************************************************/
 /*      DECLARE AF OUTPUT                                            */
 /*********************************************************************/
                                                                        
 DCL     1 BE317,                                                       
           %INCLUDE BE31700N;;                                          
                                                                        
 /*********************************************************************/
 /*      DECLARE AF TÆLLEVÆRKER                                       */
 /*********************************************************************/
                                                                        
 DCL     1 TV,                                                          
           2  SLET_UDL       FIXED DEC (7),                             
           2  SLET_DANSK     FIXED DEC (7),                             
           2  LÆSTE_IALT     FIXED DEC (7),                             
           2  SKRIV_IALT     FIXED DEC (7);                             
                                                                        
                                                                        
 /*********************************************************************/
 /*      DECLARE AF DIVERSE FELTER                                    */
 /*********************************************************************/
                                                                        
 DCL     EOF_BE316       BIT       (1);                                 
 DCL     JA              BIT       (1)       INIT ('1'B);               
 DCL     NEJ             BIT       (1)       INIT ('0'B);               
                                                                        
 DCL     UDDATO          CHAR      (10);                                
                                                                        
 DCL     DATO1           CHAR      (26);                                
                                                                        
 DCL     1 DATO2         DEF DATO1,                                     
           2 ÅR          CHAR      (04),                                
           2 F1          CHAR      (01),                                
           2 MD          CHAR      (02),                                
           2 F2          CHAR      (01),                                
           2 DG          CHAR      (02);                                
                                                                        
 /*********************************************************************/
 /*      DECLARE AF PRINTLINIER TIL KONTROLLISTE                      */
 /*********************************************************************/
                                                                        
 DCL     1 OVLIN1,                                                      
           2 CTL         CHAR      (01),                                
           2 TXT1        CHAR      (50) INIT ('PROGRAM: BE31725L'),     
           2 TXT2        CHAR      (58) INIT                            
             ('DAN SAMLET EVS UDTRÆK TIL BEREGNING AF EVS SLUT   '),    
           2 TXT3        CHAR      (14) INIT ('UDSKREVET DEN'),         
           2 DATO        CHAR      (10);                                
 DCL     1 OVLIN2,                                                      
           2 CTL         CHAR      (01),                                
           2 F1          CHAR      (51) INIT (' '),                     
           2 TXT1        CHAR      (91) INIT                            
           ('          KØRSELSKONTROLLISTE');                           
 DCL     LINIE           CHAR      (133);                               
 DCL     1  DETAIL                           DEF LINIE,                 
          2  STYR        CHAR      (1),                                 
          2  TEKST       CHAR      (70),                                
          2  ANTAL       PIC       'Z.ZZZ.ZZ9';                         
                                                                        
 DCL     1  SLET_OPL                         DEF LINIE,                 
          2  STYR        CHAR      (1),                                 
          2  DCPRNUM     PIC   '(10)9',                                 
          2  FILLER1     CHAR      (1),                                 
          2  EKOMNUM     PIC   '(03)9',                                 
          2  STREG       CHAR      (1),                                 
          2  EEJDNUM     PIC   '(06)9',                                 
          2  FILLER2     CHAR      (1),                                 
          2  EENHEDLBNUM  PIC   '(02)9',                                
          2  FILLER3     CHAR      (1),                                 
          2  TEKST       CHAR  (67);                                    
                                                                        
         %INCLUDE ZZ20301M;                                             
 /*********************************************************************/
 /*      H O U S E - K E E P I N G                                    */
 /*********************************************************************/
                                                                        
         CALL ZS61900;                                                  
                                                                        
         OPEN FILE (BE316I) TITLE ('BE31600I');                         
         OPEN FILE (BE317O) TITLE ('BE31700O');                         
                                                                        
         CALL ZZ20390 ('*OPEN','BE31701Y');                             
         CALL ZZ20390 ('*OPEN','BE31702Y');                             
                                                                        
                                                                        
         ON ENDFILE (BE316I)                                            
         BEGIN;                                                         
            EOF_BE316 = JA;                                             
         END;                                                           
                                                                        
                                                                        
         CALL ZS38100(DATO1);                                           
         UDDATO = DATO2.DG !! '/' !! DATO2.MD !! '-' !! DATO2.ÅR;       
                                                                        
         TV = '';                                                       
         BE317 = '';                                                    
                                                                        
 /*********************************************************************/
 /*      P R O C E S S                                                */
 /*********************************************************************/
                                                                        
         CALL LÆS;                                                      
                                                                        
         DO WHILE (EOF_BE316 = NEJ);                                    
            CALL BEHANDL;                                               
            CALL LÆS;                                                   
         END;                                                           
                                                                        
         CALL AFSLUT;                                                   
                                                                        
 /*********************************************************************/
 /*      LÆS SAMSORTEREDE UDTRÆK                                      */
 /*********************************************************************/
 LÆS:                                                                   
 PROC;                                                                  
                                                                        
         READ FILE (BE316I) INTO (BE316_AR);                            
                                                                        
         IF EOF_BE316 = NEJ                                             
         THEN                                                           
            TV.LÆSTE_IALT = TV.LÆSTE_IALT + 1;                          
                                                                        
 END LÆS;                                                               
 /*********************************************************************/
 /*      SKRIV BEHANDLET UDTRÆK                                       */
 /*********************************************************************/
 SKRIV: PROC;                                                           
                                                                        
         /* felt BFE og VEI */ /* KS46 */                               
         BE317.BFENR        = BE317.EVS_FORSK_EJER.BFENR;               
         BE317.VUREJENDOMID = BE317.EVS_FORSK_EJER.VUREJENDOMID;        
                                                                        
         /* Defect 248 Forskud 2020 */                                  
         IF BE317.EVS_SLUT_EJER.SIDSTE.CSLET  ^= '1'  &                 
            BE317.EVS_FORSK_EJER.CSLET        ^= '1'                    
         THEN                                                           
           DO;                                                          
              WRITE FILE (BE317O) FROM (BE317);                         
              TV.SKRIV_IALT = TV.SKRIV_IALT + 1;                        
           END;                                                         
         ELSE                                                           
           DO;                                                          
             IF BE316.EKOMNR = 999                                      
             THEN                                                       
               TV.SLET_UDL = SLET_UDL + 1;                              
             ELSE                                                       
               TV.SLET_DANSK = SLET_DANSK + 1;                          
             CALL SKRIV_SLET_LISTE;                                     
           END;                                                         
                                                                        
         BE317.EVS_SLUT_EJER = '';                                      
         BE317.EVS_SLUT_ÆGTF = '';                                      
         BE317.EVS_FORSK_EJER = '';    /* F2003 */                      
         BE317.EVS_FORSK_ÆGTF = '';    /* F2003 */                      
                                                                        
 END SKRIV;                                                             
 /*********************************************************************/
 /*      BEHANDLING AF INPUT                                          */
 /*********************************************************************/
 BEHANDL:                                                               
 PROC;                                                                  
                                                                        
         SELECT;                                                        
                                                                        
            WHEN (BE316.EKOMNR = 999 & BE316.CFRATYPE = '0')            
               /* STATUS INDIVID FOR UDENLANDSKE EJENDOMME */           
               GEM_BE311_AR = BE316_AR;                                 
                                                                        
            WHEN (BE317.DCPRNUM ^= BE316.DCPRN)                         
               DO;                                                      
                  IF BE317.DCPRNUM ^= 0   /* IKKE FØRSTE GANG */        
                  THEN                                                  
                     DO;                                                
                       CALL SKRIV;                                      
                                                                        
                       BE317.CUDLAND = ' ';                             
                       BE317.CSBSK   = ' ';                             
                     END;                                               
                  CALL FLYT_DATA;                                       
               END;                                                     
                                                                        
            WHEN (BE317.DCPRNUM       = BE316.DCPRN &                   
                  (BE317.EKOMNUM     ^= BE316.EKOMNR !                  
                   BE317.EEJDNUM     ^= BE316.EEJDNR !                  
                   BE317.EENHEDLBNUM ^= BE316.EENHEDLBNR                
                   ))                                                   
               DO;                                                      
                  /* SAMME PERSON OG NY EJENDOM */                      
                  CALL SKRIV;                                           
                  CALL FLYT_DATA;                                       
               END;                                                     
                                                                        
            OTHERWISE                                                   
               DO;                                                      
                  /* SAMME PERSON OG SAMME EJENDOM */                   
                  CALL FLYT_DATA;                                       
               END;                                                     
                                                                        
         END; /* SELECT */                                              
                                                                        
 END BEHANDL;                                                           
 /*********************************************************************/
 /*      FLYT OPLYSNINGER FRA INPUT TIL RELEVANT PLADS I OUTPUT       */
 /*********************************************************************/
 FLYT_DATA:                                                             
 PROC;                                                                  
                                                                        
         BE317.DCPRNUM = BE316.DCPRN;                                   
         BE317.EKOMNUM = BE316.EKOMNR;                                  
         BE317.EEJDNUM = BE316.EEJDNR;                                  
         BE317.EENHEDLBNUM = BE316.EENHEDLBNR;                          
         BE317.CITYP   = 0;                                             
                                                                        
         SELECT;                                                        
                                                                        
           WHEN (BE317.EKOMNUM = 999)          /* UDENLANDSK EJENDOM  */
             DO;                                                        
                CALL OPSÆT_STATUS;                                      
                IF BE316.CFRATYPE = '2'                                 
                THEN                                                    
                   BE317.EVS_SLUT_EJER  = BE300, BY NAME;               
                ELSE                                                    
                   BE317.EVS_FORSK_EJER = BE301, BY NAME;   /* F2003 */ 
             END;                                                       
                                                                        
           WHEN (BE316.CFRATYPE = '1')         /* DUBLET FRA EVS SLUT */
            DO;                                                         
             BE317.EVS_SLUT_ÆGTF = BE315.EVS_SLUT, BY NAME;             
             BE317.CSBSK = BE315.CSBSK;                                 
            END;                                                        
                                                                        
           WHEN (BE316.CFRATYPE = '2')          /* EJER FRA EVS SLUT */ 
            DO;                                                         
             BE317.EVS_SLUT_EJER = BE315.EVS_SLUT, BY NAME;             
             BE317.CSBSK = BE315.CSBSK;                                 
            END;                                                        
                                                                        
           WHEN (BE316.CFRATYPE = '3')          /* DUBLET FRA FORSKUD */
            DO;                                                         
             BE317.EVS_FORSK_ÆGTF = BE302.EVS_FORSKUD, BY NAME;         
            END;                                                        
                                                                        
           WHEN (BE316.CFRATYPE = '4')          /* EJER FRA FORSKUD */  
            DO;                                                         
             IF BE302.COMPLA ^= '1'         /* IKKE HVIS OMPLACERET */  
             THEN                                                       
                BE317.EVS_FORSK_EJER = BE302.EVS_FORSKUD, BY NAME;      
            END;                                                        
                                                                        
           OTHERWISE;                                                   
                                                                        
         END; /* SELECT */                                              
                                                                        
 END FLYT_DATA;                                                         
 /*********************************************************************/
 /*      OPSÆT STATUS KODE FOR EN PERSONS UDENLANDSKE EJENDOMME       */
 /*********************************************************************/
 OPSÆT_STATUS:                                                          
 PROC;                                                                  
                                                                        
         SELECT;                                                        
                                                                        
           WHEN (GEM_BE311.FEJET_IALT = GEM_BE311.FEJET_SLUT &          
                 GEM_BE311.FEJET_FORSKU = 0)                            
              DO;                                                       
                 /* ALLE EJES KUN I SLUT */                             
                 BE317.CUDLAND = '1';                                   
              END;                                                      
                                                                        
           WHEN (GEM_BE311.FEJET_IALT = GEM_BE311.FEJET_FORSKU &        
                 GEM_BE311.FEJET_SLUT = 0)                              
              DO;                                                       
                 /* ALLE EJES KUN I FORSKUD */                          
                 BE317.CUDLAND = '2';                                   
              END;                                                      
                                                                        
           WHEN (GEM_BE311.FEJET_IALT = GEM_BE311.FEJET_BÅDEOG &        
                 GEM_BE311.FEJET_IALT = GEM_BE311.FEJET_SLUT &          
                 GEM_BE311.FEJET_IALT = GEM_BE311.FEJET_FORSKU)         
              DO;                                                       
                 BE317.CUDLAND = '3';                                   
                 /* ALLE EJES BÅDE I SLUT OG FORSKUD */                 
              END;                                                      
                                                                        
           OTHERWISE                                                    
              BE317.CUDLAND = '4';                                      
                                                                        
         END; /* SELECT */                                              
                                                                        
 END OPSÆT_STATUS;                                                      
 /*********************************************************************/
 /*      UDSKRIV KØRSELSKONTROLLISTE                                  */
 /*********************************************************************/
 KONTROLLISTE:                                                          
 PROC;                                                                  
                                                                        
         LINIE = ZZIK1;                                                 
         CALL ZZ20300 (LINIE,'01');                                     
                                                                        
         OVLIN1.CTL= ZZWS1;                                             
         OVLIN1.DATO = UDDATO;                                          
         CALL ZZ20300 (OVLIN1,'01');                                    
                                                                        
         OVLIN2.CTL= ZZWS2;                                             
         CALL ZZ20300 (OVLIN2,'01');                                    
                                                                        
         LINIE = '';                                                    
         DETAIL.STYR = ZZWS2;                                           
         DETAIL.TEKST = 'ANTAL LÆSTE IALT                    B.H31600D';
         DETAIL.ANTAL = TV.LÆSTE_IALT;                                  
         CALL ZZ20300 (LINIE,'01');                                     
                                                                        
         LINIE = '';                                                    
         DETAIL.STYR = ZZWS2;                                           
         DETAIL.TEKST = 'ANTAL SLETTET UDLAND                         ';
         DETAIL.ANTAL = TV.SLET_UDL;                                    
         CALL ZZ20300 (LINIE,'01');                                     
                                                                        
         LINIE = '';                                                    
         DETAIL.STYR = ZZWS2;                                           
         DETAIL.TEKST = 'ANTAL SLETTET DANSKE                         ';
         DETAIL.ANTAL = TV.SLET_DANSK;                                  
         CALL ZZ20300 (LINIE,'01');                                     
                                                                        
         LINIE = '';                                                    
         DETAIL.STYR = ZZWS2;                                           
         DETAIL.TEKST = 'ANTAL SKREVET IALT                  B.H31700D';
         DETAIL.ANTAL = TV.SKRIV_IALT;                                  
         CALL ZZ20300 (LINIE,'01');                                     
                                                                        
                                                                        
                                                                        
 END KONTROLLISTE;                                                      
 /*********************************************************************/
 /*      UDSKRIV SLETTE LISTE                                         */
 /*********************************************************************/
 SKRIV_SLET_LISTE: PROC;                                                
                                                                        
         LINIE = '';                                                    
         LINIE = ZZIK1;                                                 
         CALL ZZ20300 (LINIE,'02');                                     
                                                                        
         LINIE = '';                                                    
         SLET_OPL.STYR = ZZWS2;                                         
         SLET_OPL.DCPRNUM = BE317.DCPRNUM;                              
         SLET_OPL.EKOMNUM = BE317.EKOMNUM;                              
         SLET_OPL.STREG   = '-';                                        
         SLET_OPL.EEJDNUM = BE317.EEJDNUM;                              
         SLET_OPL.EENHEDLBNUM = BE317.EENHEDLBNUM;                      
                                                                        
         IF BE317.EVS_SLUT_EJER.SIDSTE.CSLET  = '1' &                   
            BE317.EVS_FORSK_EJER.CSLET = '1'                            
         THEN                                                           
           SLET_OPL.TEKST   = 'SLETTET I SLUT OG FORSKUD ';             
         ELSE                                                           
           IF BE317.EVS_SLUT_EJER.SIDSTE.CSLET  = '1'                   
           THEN                                                         
             SLET_OPL.TEKST   = 'SLETTET I SLUT';                       
           ELSE                                                         
             IF BE317.EVS_FORSK_EJER.CSLET = '1'                        
             THEN                                                       
               SLET_OPL.TEKST   = 'SLETTET I FORSKUD';                  
                                                                        
         CALL ZZ20300 (LINIE,'02');                                     
                                                                        
                                                                        
 END SKRIV_SLET_LISTE;                                                  
 /*********************************************************************/
 /*      AFSLUTNING                                                   */
 /*********************************************************************/
 AFSLUT:                                                                
 PROC;                                                                  
                                                                        
         CALL SKRIV; /* SIDSTE EJERS SIDSTE INDIVID */                  
                                                                        
         CALL KONTROLLISTE;                                             
                                                                        
         CALL ZZ20300(' ','99');                                        
                                                                        
         CLOSE FILE (BE316I),                                           
               FILE (BE317O);                                           
                                                                        
 END AFSLUT;                                                            
                                                                        
 END BE31725;
