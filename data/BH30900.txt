   /* PUT SKIP LIST('NØGLE_2001 :',NØGLE_2001); */                      
   /* PUT SKIP LIST('NØGLE_2000 :',NØGLE_2000); */

             NØGLE_2001 >= NØGLE_2000);                                 
     IF UDTRÆK_SLUT_2001.DCPRN  = UDTRÆK_SLUT_2000.DCPRN  &             
        UDTRÆK_SLUT_2001.EKOMNR = UDTRÆK_SLUT_2000.EKOMNR &             
        UDTRÆK_SLUT_2001.EEJDNR = UDTRÆK_SLUT_2000.EEJDNR

         STATUS = '1';                                                  
   /*    PUT SKIP LIST ('PERSON FUNDET PÅ FIL BH3900I'); */             
       END;                                                             
     ELSE                                                               
       DO;                                                              
         READ FILE (BH3900I) INTO (UDTRÆK_SLUT_2000);                   
         UD_2000.PERSONNR = UDTRÆK_SLUT_2000.DCPRN;                     
         UD_2000.KOMUNENR = UDTRÆK_SLUT_2000.EKOMNR;                    
         UD_2000.EJENDONR = UDTRÆK_SLUT_2000.EEJDNR;                    
         /* PUT SKIP LIST('NØGLE_2000',NØGLE_2000); */                  
       END;

    IF STATUS ^= '1' THEN ;                                             
   /*  PUT SKIP LIST ('PERSON IKKE FUNDET PÅ FIL BH3900I'); */          
 END SKAF_DATOER_DAN;

          (UDTRÆK_SLUT_2000.SIDSTE.CEJBEVS = '5'      !                 
           UDTRÆK_SLUT_2000.EKOMNR = 999)             &

    /*     PUT SKIP LIST('DOVTG INDSAT F ',UDTRÆK_SLUT_2001.DCPRN); */  
         END;                                                           
                                                                        
                                                                        
       IF UDTRÆK_SLUT_2001.SIDSTE.DOVTG < 20010101    &                 
          (UDTRÆK_SLUT_2000.SIDSTE.CEJBEVS = '5'      !                 
           UDTRÆK_SLUT_2000.EKOMNR = 999)             &                 
           UDTRÆK_SLUT_2000.SIDSTE.DAFSTAA = 0        &                 
           UDTRÆK_SLUT_2001.FØRSTE.DOVTG =                              
                            UDTRÆK_SLUT_2001.SIDSTE.DOVTG

           UDTRÆK_SLUT_2001.SIDSTE.DOVTG =                              
                              UDTRÆK_SLUT_2000.SIDSTE.DOVTG;
